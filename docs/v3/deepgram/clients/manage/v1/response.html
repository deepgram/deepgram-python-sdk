 11@dataclass
 12class BaseResponse(DataClassJsonMixin):
 13    """
 14    BaseResponse class used to define the common methods and properties for all response classes.
 15    """
 16
 17    def __getitem__(self, key):
 18        _dict = self.to_dict()
 19        return _dict[key]
 20
 21    def __setitem__(self, key, val):
 22        self.__dict__[key] = val
 23
 24    def __str__(self) -> str:
 25        return self.to_json(indent=4)
 26
 27    def eval(self, key: str) -> str:
 28        """
 29        This method is used to evaluate a key in the response object using a dot notation style method.
 30        """
 31        keys = key.split(".")
 32        result: Dict[Any, Any] = self.to_dict()
 33        for k in keys:
 34            if isinstance(result, dict) and k in result:
 35                result = result[k]
 36            elif isinstance(result, list) and k.isdigit() and int(k) < len(result):
 37                result = result[int(k)]
 38            else:
 39                return ""
 40        return str(result)
