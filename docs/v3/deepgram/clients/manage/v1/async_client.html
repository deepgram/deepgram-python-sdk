 41class AsyncManageClient(AbstractAsyncRestClient):
 42    """
 43    A client for managing Deepgram projects and associated resources via the Deepgram API.
 44
 45    This class provides methods for performing various operations on Deepgram projects, including:
 46    - Retrieving project details
 47    - Updating project settings
 48    - Managing project members and scopes
 49    - Handling project invitations
 50    - Monitoring project usage and balances
 51
 52    Args:
 53        config (DeepgramClientOptions): all the options for the client.
 54    """
