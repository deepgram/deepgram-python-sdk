203    async def _emit(self, event: SpeakWebSocketEvents, *args, **kwargs) -> None:
204        """
205        Emits events to the registered event handlers.
206        """
207        self._logger.debug("AsyncSpeakStreamClient._emit ENTER")
208        self._logger.debug("callback handlers for: %s", event)
209
210        # debug the threads
211        for thread in threading.enumerate():
212            self._logger.debug("after running thread: %s", thread.name)
213        self._logger.debug("number of active threads: %s", threading.active_count())
214
215        tasks = []
216        for handler in self._event_handlers[event]:
217            task = asyncio.create_task(handler(self, *args, **kwargs))
218            tasks.append(task)
219
220        if tasks:
221            self._logger.debug("waiting for tasks to finish...")
222            await asyncio.gather(*tasks, return_exceptions=True)
223            tasks.clear()

225        # debug the threads
226        for thread in threading.enumerate():
227            self._logger.debug("after running thread: %s", thread.name)
228        self._logger.debug("number of active threads: %s", threading.active_count())
229

233    async def _listening(self) -> None:
234        """
235        Listens for messages from the WebSocket connection.
236        """
237        self._logger.debug("AsyncSpeakStreamClient._listening ENTER")
238
239        while True:
240            try:
241                if self._exit_event.is_set():
242                    self._logger.notice("_listening exiting gracefully")
243                    self._logger.debug("AsyncSpeakStreamClient._listening LEAVE")
244                    return
245
246                if self._socket is None:
247                    self._logger.warning("socket is empty")
248                    self._logger.debug("AsyncSpeakStreamClient._listening LEAVE")
249                    return
250
251                message = await self._socket.recv()
252
253                if message is None:
254                    self._logger.spam("message is None")
255                    continue
256
257                if isinstance(message, bytes):
258                    self._logger.debug("Binary data received")
259                    await self._emit(
260                        SpeakWebSocketEvents(SpeakWebSocketEvents.AudioData),
261                        data=message,
262                        **dict(cast(Dict[Any, Any], self._kwargs)),
263                    )
264                else:
265                    data = json.loads(message)
266                    response_type = data.get("type")
267                    self._logger.debug(
268                        "response_type: %s, data: %s", response_type, data
269                    )
270
271                    match response_type:
272                        case SpeakWebSocketEvents.Open:
273                            open_result: OpenResponse = OpenResponse.from_json(message)
274                            self._logger.verbose("OpenResponse: %s", open_result)
275                            await self._emit(
276                                SpeakWebSocketEvents(SpeakWebSocketEvents.Open),
277                                open=open_result,
278                                **dict(cast(Dict[Any, Any], self._kwargs)),
279                            )
280                        case SpeakWebSocketEvents.Metadata:
281                            meta_result: MetadataResponse = MetadataResponse.from_json(
282                                message
283                            )
284                            self._logger.verbose("MetadataResponse: %s", meta_result)
285                            await self._emit(
286                                SpeakWebSocketEvents(SpeakWebSocketEvents.Metadata),
287                                metadata=meta_result,
288                                **dict(cast(Dict[Any, Any], self._kwargs)),
289                            )
290                        case SpeakWebSocketEvents.Flush:
291                            fl_result: FlushedResponse = FlushedResponse.from_json(
292                                message
293                            )
294                            self._logger.verbose("FlushedResponse: %s", fl_result)
295                            await self._emit(
296                                SpeakWebSocketEvents(SpeakWebSocketEvents.Flush),
297                                flushed=fl_result,
298                                **dict(cast(Dict[Any, Any], self._kwargs)),
299                            )
300                        case SpeakWebSocketEvents.Close:
301                            close_result: CloseResponse = CloseResponse.from_json(
302                                message
303                            )
304                            self._logger.verbose("CloseResponse: %s", close_result)
305                            await self._emit(
306                                SpeakWebSocketEvents(SpeakWebSocketEvents.Close),
307                                close=close_result,
308                                **dict(cast(Dict[Any, Any], self._kwargs)),
309                            )
310                        case SpeakWebSocketEvents.Warning:
311                            war_warning: WarningResponse = WarningResponse.from_json(
312                                message
313                            )
314                            self._logger.verbose("WarningResponse: %s", war_warning)
315                            await self._emit(
316                                SpeakWebSocketEvents(SpeakWebSocketEvents.Warning),
317                                warning=war_warning,
318                                **dict(cast(Dict[Any, Any], self._kwargs)),
319                            )
320                        case SpeakWebSocketEvents.Error:
321                            err_error: ErrorResponse = ErrorResponse.from_json(message)
322                            self._logger.verbose("ErrorResponse: %s", err_error)
323                            await self._emit(
324                                SpeakWebSocketEvents(SpeakWebSocketEvents.Error),
325                                error=err_error,
326                                **dict(cast(Dict[Any, Any], self._kwargs)),
327                            )
328                        case _:
329                            self._logger.warning(
330                                "Unknown Message: response_type: %s, data: %s",
331                                response_type,
332                                data,
333                            )
334                            unhandled_error: UnhandledResponse = UnhandledResponse(
335                                type=SpeakWebSocketEvents(
336                                    SpeakWebSocketEvents.Unhandled
337                                ),
338                                raw=message,
339                            )
340                            await self._emit(
341                                SpeakWebSocketEvents(SpeakWebSocketEvents.Unhandled),
342                                unhandled=unhandled_error,
343                                **dict(cast(Dict[Any, Any], self._kwargs)),
344                            )
345
346            except websockets.exceptions.ConnectionClosedOK as e:
347                self._logger.notice(f"_listening({e.code}) exiting gracefully")
348                self._logger.debug("AsyncSpeakStreamClient._listening LEAVE")
349                return
350
351            except websockets.exceptions.ConnectionClosed as e:
352                if e.code == 1000:
353                    self._logger.notice(f"_listening({e.code}) exiting gracefully")
354                    self._logger.debug("AsyncSpeakStreamClient._listening LEAVE")
355                    return
356
357                self._logger.error(
358                    "ConnectionClosed in AsyncSpeakStreamClient._listening with code %s: %s",
359                    e.code,
360                    e.reason,
361                )
362                cc_error: ErrorResponse = ErrorResponse(
363                    "ConnectionClosed in AsyncSpeakStreamClient._listening",
364                    f"{e}",
365                    "ConnectionClosed",
366                )
367                await self._emit(
368                    SpeakWebSocketEvents(SpeakWebSocketEvents.Error),
369                    error=cc_error,
370                    **dict(cast(Dict[Any, Any], self._kwargs)),
371                )
372
373                # signal exit and close
374                await self._signal_exit()
375
376                self._logger.debug("AsyncSpeakStreamClient._listening LEAVE")
377
378                if self._config.options.get("termination_exception") == "true":
379                    raise
380                return
381
382            except websockets.exceptions.WebSocketException as e:
383                self._logger.error(
384                    "WebSocketException in AsyncSpeakStreamClient._listening: %s", e
385                )
386                ws_error: ErrorResponse = ErrorResponse(
387                    "WebSocketException in AsyncSpeakStreamClient._listening",
388                    f"{e}",
389                    "WebSocketException",
390                )
391                await self._emit(
392                    SpeakWebSocketEvents(SpeakWebSocketEvents.Error),
393                    error=ws_error,
394                    **dict(cast(Dict[Any, Any], self._kwargs)),
395                )
396
397                # signal exit and close
398                await self._signal_exit()
399
400                self._logger.debug("AsyncSpeakStreamClient._listening LEAVE")
401
402                if self._config.options.get("termination_exception") == "true":
403                    raise
404                return
405
406            except Exception as e:  # pylint: disable=broad-except
407                self._logger.error(
408                    "Exception in AsyncSpeakStreamClient._listening: %s", e
409                )
410                e_error: ErrorResponse = ErrorResponse(
411                    "Exception in AsyncSpeakStreamClient._listening",
412                    f"{e}",
413                    "Exception",
414                )
415                await self._emit(
416                    SpeakWebSocketEvents(SpeakWebSocketEvents.Error),
417                    error=e_error,
418                    **dict(cast(Dict[Any, Any], self._kwargs)),
419                )
420
421                # signal exit and close
422                await self._signal_exit()
423
424                self._logger.debug("AsyncSpeakStreamClient._listening LEAVE")
425
426                if self._config.options.get("termination_exception") == "true":
427                    raise
428                return

433        """
434        Sends data over the WebSocket connection.
435        """
436        self._logger.spam("AsyncSpeakStreamClient.send ENTER")

439            self._logger.notice("send exiting gracefully")
440            self._logger.debug("AsyncSpeakStreamClient.send LEAVE")
441            return False
442
443        if self._socket is not None:
444            try:
445                await self._socket.send(
446                    json.dumps({"type": "Speak", "text": text_input})
447                )
448            except websockets.exceptions.ConnectionClosedOK as e:
449                self._logger.notice(f"send() exiting gracefully: {e.code}")

452                    raise
453                return True
454            except websockets.exceptions.ConnectionClosed as e:
455                if e.code in [1000, 1001]:
456                    self._logger.notice(f"send({e.code}) exiting gracefully")
457                    self._logger.debug("AsyncSpeakStreamClient.send LEAVE")
458                    if self._config.options.get("termination_exception_send") == "true":
459                        raise
460                    return True
461
462                self._logger.error("send() failed - ConnectionClosed: %s", str(e))
463                self._logger.spam("AsyncSpeakStreamClient.send LEAVE")
464                if self._config.options.get("termination_exception_send") == "true":
465                    raise
466                return False
467            except websockets.exceptions.WebSocketException as e:
468                self._logger.error("send() failed - WebSocketException: %s", str(e))
469                self._logger.spam("AsyncSpeakStreamClient.send LEAVE")
470                if self._config.options.get("termination_exception_send") == "true":
471                    raise
472                return False
473            except Exception as e:  # pylint: disable=broad-except
474                self._logger.error("send() failed - Exception: %s", str(e))
475                self._logger.spam("AsyncSpeakStreamClient.send LEAVE")
476                if self._config.options.get("termination_exception_send") == "true":
477                    raise
478                return False
479
480            self._logger.spam("send() succeeded")
481            self._logger.spam("AsyncSpeakStreamClient.send LEAVE")
482            return True
483
484        self._logger.spam("send() failed. socket is None")
485        self._logger.spam("AsyncSpeakStreamClient.send LEAVE")

521        Closes the WebSocket connection gracefully.
522        """
523        self._logger.debug("AsyncSpeakStreamClient.finish ENTER")
524
525        # signal exit
526        await self._signal_exit()
527
528        # stop the threads
529        self._logger.verbose("cancelling tasks...")
530        try:
531            # Before cancelling, check if the tasks were created
532            # debug the threads
533            for thread in threading.enumerate():
534                self._logger.debug("before running thread: %s", thread.name)
535            self._logger.debug("number of active threads: %s", threading.active_count())
536
537            tasks = []
538
539            if self._listen_thread is not None:
540                self._listen_thread.cancel()
541                tasks.append(self._listen_thread)
542                self._logger.notice("processing _listen_thread cancel...")
543
544            # Use asyncio.gather to wait for tasks to be cancelled
545            await asyncio.wait_for(
546                asyncio.gather(*tasks), timeout=10
547            )  # Prevent indefinite waiting
548            self._logger.notice("threads joined")
549
550            # debug the threads
551            for thread in threading.enumerate():
552                self._logger.debug("after running thread: %s", thread.name)
553            self._logger.debug("number of active threads: %s", threading.active_count())
554
555            self._logger.notice("finish succeeded")
556            self._logger.spam("AsyncSpeakStreamClient.finish LEAVE")
557            return True
558
559        except asyncio.CancelledError as e:
560            self._logger.error("tasks cancelled error: %s", e)
561            self._logger.debug("AsyncSpeakStreamClient.finish LEAVE")
562            return False
563
564        except asyncio.TimeoutError as e:

570        # send close event
571        self._logger.verbose("closing socket...")
572        if self._socket is not None:
573            self._logger.verbose("send CloseStream...")
574            try:
575                # if the socket connection is closed, the following line might throw an error
576                await self.send(json.dumps({"type": "CloseStream"}))
577            except websockets.exceptions.ConnectionClosedOK as e:
578                self._logger.notice("_signal_exit  - ConnectionClosedOK: %s", e.code)
579            except websockets.exceptions.ConnectionClosed as e:
580                self._logger.notice("_signal_exit  - ConnectionClosed: %s", e.code)
581            except websockets.exceptions.WebSocketException as e:
582                self._logger.error("_signal_exit - WebSocketException: %s", str(e))
583            except Exception as e:  # pylint: disable=broad-except
584                self._logger.error("_signal_exit - Exception: %s", str(e))
585
586            # push close event
587            try:
588                await self._emit(
589                    SpeakWebSocketEvents(SpeakWebSocketEvents.Close),
590                    close=CloseResponse(type=SpeakWebSocketEvents.Close),
591                    **dict(cast(Dict[Any, Any], self._kwargs)),
592                )
593            except Exception as e:  # pylint: disable=broad-except
594                self._logger.error("_emit - Exception: %s", e)
595
596            # wait for task to send
597            await asyncio.sleep(0.5)
598
599        # signal exit
600        self._exit_event.set()
601
602        # closes the WebSocket connection gracefully
603        self._logger.verbose("clean up socket...")
604        if self._socket is not None:
605            self._logger.verbose("socket.wait_closed...")
606            try:
607                await self._socket.close()
608            except websockets.exceptions.WebSocketException as e:
609                self._logger.error("socket.wait_closed failed: %s", e)
610
