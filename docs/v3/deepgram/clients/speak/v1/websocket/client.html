 83        self,
 84        options: Optional[Union[SpeakOptions, Dict]] = None,
 85        addons: Optional[Dict] = None,
 86        headers: Optional[Dict] = None,
 87        members: Optional[Dict] = None,
 88        **kwargs,
 89    ) -> bool:
 90        """
 91        Starts the WebSocket connection for text-to-speech synthesis.
 92        """
 93        self._logger.debug("SpeakStreamClient.start ENTER")
 94        self._logger.info("options: %s", options)
 95        self._logger.info("addons: %s", addons)
 96        self._logger.info("headers: %s", headers)
 97        self._logger.info("members: %s", members)
 98        self._logger.info("kwargs: %s", kwargs)
 99
100        if isinstance(options, SpeakOptions) and not options.check():
101            self._logger.error("options.check failed")
102            self._logger.debug("SpeakStreamClient.start LEAVE")
103            raise DeepgramError("Fatal text-to-speech options error")
104
105        self._addons = addons
106        self._headers = headers
107
108        # add "members" as members of the class
109        if members is not None:
110            self.__dict__.update(members)
111
112        # set kwargs as members of the class
113        if kwargs is not None:
114            self._kwargs = kwargs
115        else:
116            self._kwargs = {}
117
118        if isinstance(options, SpeakOptions):
119            self._logger.info("SpeakOptions switching class -> dict")
120            self._options = options.to_dict()
121        elif options is not None:
122            self._options = options
123        else:
124            self._options = {}
125
126        combined_options = self._options
127        if self._addons is not None:
128            self._logger.info("merging addons to options")
129            combined_options.update(self._addons)
130            self._logger.info("new options: %s", combined_options)
131        self._logger.debug("combined_options: %s", combined_options)
132
133        combined_headers = self._config.headers
134        if self._headers is not None:
135            self._logger.info("merging headers to options")
136            combined_headers.update(self._headers)
137            self._logger.info("new headers: %s", combined_headers)
138        self._logger.debug("combined_headers: %s", combined_headers)
139
140        url_with_params = append_query_params(self._websocket_url, combined_options)
141        try:
142            self._socket = connect(url_with_params, additional_headers=combined_headers)
143            self._exit_event.clear()
144
145            # debug the threads
146            for thread in threading.enumerate():
147                self._logger.debug("after running thread: %s", thread.name)
148            self._logger.debug("number of active threads: %s", threading.active_count())
149
150            # listening thread
151            self._listen_thread = threading.Thread(target=self._listening)
152            self._listen_thread.start()
153
154            # debug the threads
155            for thread in threading.enumerate():
156                self._logger.debug("after running thread: %s", thread.name)
157            self._logger.debug("number of active threads: %s", threading.active_count())
158
159            # push open event
160            self._emit(
161                SpeakWebSocketEvents(SpeakWebSocketEvents.Open),
162                OpenResponse(type=SpeakWebSocketEvents.Open),
163            )
164
165            self._logger.notice("start succeeded")
166            self._logger.debug("SpeakStreamClient.start LEAVE")
167            return True
168        except websockets.ConnectionClosed as e:
169            self._logger.error("ConnectionClosed in SpeakStreamClient.start: %s", e)
170            self._logger.debug("SpeakStreamClient.start LEAVE")
171            if self._config.options.get("termination_exception_connect") == "true":
172                raise e
173            return False
174        except websockets.exceptions.WebSocketException as e:
175            self._logger.error("WebSocketException in SpeakStreamClient.start: %s", e)
176            self._logger.debug("SpeakStreamClient.start LEAVE")
177            if self._config.options.get("termination_exception_connect") == "true":
178                raise e
179            return False
180        except Exception as e:  # pylint: disable=broad-except
181            self._logger.error("WebSocketException in SpeakStreamClient.start: %s", e)
182            self._logger.debug("SpeakStreamClient.start LEAVE")
183            if self._config.options.get("termination_exception_connect") == "true":
184                raise e
185            return False

199    def _emit(self, event: SpeakWebSocketEvents, *args, **kwargs) -> None:
200        """

203        self._logger.debug("callback handlers for: %s", event)
204        for handler in self._event_handlers[event]:
205            handler(self, *args, **kwargs)

208    def _listening(
209        self,
210    ) -> None:
211        """
212        Listens for messages from the WebSocket connection.
213        """
214        self._logger.debug("SpeakStreamClient._listening ENTER")
215
216        while True:
217            try:
218                if self._exit_event.is_set():
219                    self._logger.notice("_listening exiting gracefully")
220                    self._logger.debug("SpeakStreamClient._listening LEAVE")
221                    return
222
223                if self._socket is None:
224                    self._logger.warning("socket is empty")
225                    self._logger.debug("SpeakStreamClient._listening LEAVE")
226                    return
227
228                message = self._socket.recv()
229
230                if message is None:
231                    self._logger.info("message is empty")
232                    continue
233
234                if isinstance(message, bytes):
235                    self._logger.debug("Binary data received")
236                    self._emit(
237                        SpeakWebSocketEvents(SpeakWebSocketEvents.AudioData),
238                        data=message,
239                        **dict(cast(Dict[Any, Any], self._kwargs)),
240                    )
241                else:
242                    data = json.loads(message)
243                    response_type = data.get("type")
244                    self._logger.debug(
245                        "response_type: %s, data: %s", response_type, data
246                    )
247
248                    match response_type:
249                        case SpeakWebSocketEvents.Open:
250                            open_result: OpenResponse = OpenResponse.from_json(message)
251                            self._logger.verbose("OpenResponse: %s", open_result)
252                            self._emit(
253                                SpeakWebSocketEvents(SpeakWebSocketEvents.Open),
254                                open=open_result,
255                                **dict(cast(Dict[Any, Any], self._kwargs)),
256                            )
257                        case SpeakWebSocketEvents.Metadata:
258                            meta_result: MetadataResponse = MetadataResponse.from_json(
259                                message
260                            )
261                            self._logger.verbose("MetadataResponse: %s", meta_result)
262                            self._emit(
263                                SpeakWebSocketEvents(SpeakWebSocketEvents.Metadata),
264                                metadata=meta_result,
265                                **dict(cast(Dict[Any, Any], self._kwargs)),
266                            )
267                        case SpeakWebSocketEvents.Flush:
268                            fl_result: FlushedResponse = FlushedResponse.from_json(
269                                message
270                            )
271                            self._logger.verbose("FlushedResponse: %s", fl_result)
272                            self._emit(
273                                SpeakWebSocketEvents(SpeakWebSocketEvents.Flush),
274                                flushed=fl_result,
275                                **dict(cast(Dict[Any, Any], self._kwargs)),
276                            )
277                        case SpeakWebSocketEvents.Close:
278                            close_result: CloseResponse = CloseResponse.from_json(
279                                message
280                            )
281                            self._logger.verbose("CloseResponse: %s", close_result)
282                            self._emit(
283                                SpeakWebSocketEvents(SpeakWebSocketEvents.Close),
284                                close=close_result,
285                                **dict(cast(Dict[Any, Any], self._kwargs)),
286                            )
287                        case SpeakWebSocketEvents.Warning:
288                            war_warning: WarningResponse = WarningResponse.from_json(
289                                message
290                            )
291                            self._logger.verbose("WarningResponse: %s", war_warning)
292                            self._emit(
293                                SpeakWebSocketEvents(SpeakWebSocketEvents.Warning),
294                                warning=war_warning,
295                                **dict(cast(Dict[Any, Any], self._kwargs)),
296                            )
297                        case SpeakWebSocketEvents.Error:
298                            err_error: ErrorResponse = ErrorResponse.from_json(message)
299                            self._logger.verbose("ErrorResponse: %s", err_error)
300                            self._emit(
301                                SpeakWebSocketEvents(SpeakWebSocketEvents.Error),
302                                error=err_error,
303                                **dict(cast(Dict[Any, Any], self._kwargs)),
304                            )
305                        case _:
306                            self._logger.warning(
307                                "Unknown Message: response_type: %s, data: %s",
308                                response_type,
309                                data,
310                            )
311                            unhandled_error: UnhandledResponse = UnhandledResponse(
312                                type=SpeakWebSocketEvents(
313                                    SpeakWebSocketEvents.Unhandled
314                                ),
315                                raw=message,
316                            )
317                            self._emit(
318                                SpeakWebSocketEvents(SpeakWebSocketEvents.Unhandled),
319                                unhandled=unhandled_error,
320                                **dict(cast(Dict[Any, Any], self._kwargs)),
321                            )
322
323            except websockets.exceptions.ConnectionClosedOK as e:
324                self._logger.notice(f"_listening({e.code}) exiting gracefully")
325                self._logger.debug("SpeakStreamClient._listening LEAVE")
326                return
327
328            except websockets.exceptions.ConnectionClosed as e:
329                if e.code == 1000:
330                    self._logger.notice(f"_listening({e.code}) exiting gracefully")
331                    self._logger.debug("SpeakStreamClient._listening LEAVE")
332                    return
333
334                self._logger.error(
335                    "ConnectionClosed in SpeakStreamClient._listening with code %s: %s",
336                    e.code,
337                    e.reason,
338                )
339                cc_error: ErrorResponse = ErrorResponse(
340                    "ConnectionClosed in SpeakStreamClient._listening",
341                    f"{e}",
342                    "ConnectionClosed",
343                )
344                self._emit(SpeakWebSocketEvents(SpeakWebSocketEvents.Error), cc_error)
345
346                # signal exit and close
347                self._signal_exit()
348
349                self._logger.debug("SpeakStreamClient._listening LEAVE")
350
351                if self._config.options.get("termination_exception") == "true":
352                    raise
353                return
354
355            except websockets.exceptions.WebSocketException as e:
356                self._logger.error(
357                    "WebSocketException in SpeakStreamClient._listening with: %s", e
358                )
359                ws_error: ErrorResponse = ErrorResponse(
360                    "WebSocketException in SpeakStreamClient._listening",
361                    f"{e}",
362                    "WebSocketException",
363                )
364                self._emit(SpeakWebSocketEvents(SpeakWebSocketEvents.Error), ws_error)
365
366                # signal exit and close
367                self._signal_exit()
368
369                self._logger.debug("SpeakStreamClient._listening LEAVE")
370
371                if self._config.options.get("termination_exception") == "true":
372                    raise
373                return
374
375            except Exception as e:  # pylint: disable=broad-except
376                self._logger.error("Exception in SpeakStreamClient._listening: %s", e)
377                e_error: ErrorResponse = ErrorResponse(
378                    "Exception in SpeakStreamClient._listening",
379                    f"{e}",
380                    "Exception",
381                )
382                self._logger.error(
383                    "Exception in SpeakStreamClient._listening: %s", str(e)
384                )
385                self._emit(SpeakWebSocketEvents(SpeakWebSocketEvents.Error), e_error)
386
387                # signal exit and close
388                self._signal_exit()
389
390                self._logger.debug("SpeakStreamClient._listening LEAVE")
391
392                if self._config.options.get("termination_exception") == "true":
393                    raise
394                return

397    def send(self, text_input: str) -> bool:
398        """
399        Sends data over the WebSocket connection.
400        """
401        self._logger.spam("SpeakStreamClient.send ENTER")
402
403        if self._exit_event.is_set():
404            self._logger.notice("send exiting gracefully")
405            self._logger.debug("SpeakStreamClient.send LEAVE")
406            return False
407
408        if self._socket is not None:
409            with self._lock_send:
410                try:
411                    self._socket.send(json.dumps({"type": "Speak", "text": text_input}))
412                except websockets.exceptions.ConnectionClosedOK as e:
413                    self._logger.notice(f"send() exiting gracefully: {e.code}")
414                    self._logger.debug("SpeakStreamClient._keep_alive LEAVE")
415                    if self._config.options.get("termination_exception_send") == "true":
416                        raise
417                    return True
418                except websockets.exceptions.ConnectionClosed as e:
419                    if e.code == 1000:
420                        self._logger.notice(f"send({e.code}) exiting gracefully")
421                        self._logger.debug("SpeakStreamClient.send LEAVE")
422                        if (
423                            self._config.options.get("termination_exception_send")
424                            == "true"
425                        ):
426                            raise
427                        return True
428                    self._logger.error("send() failed - ConnectionClosed: %s", str(e))
429                    self._logger.spam("SpeakStreamClient.send LEAVE")
430                    if self._config.options.get("termination_exception_send") == "true":
431                        raise
432                    return False
433                except websockets.exceptions.WebSocketException as e:
434                    self._logger.error("send() failed - WebSocketException: %s", str(e))
435                    self._logger.spam("SpeakStreamClient.send LEAVE")
436                    if self._config.options.get("termination_exception_send") == "true":
437                        raise
438                    return False
439                except Exception as e:  # pylint: disable=broad-except
440                    self._logger.error("send() failed - Exception: %s", str(e))
441                    self._logger.spam("SpeakStreamClient.send LEAVE")
442                    if self._config.options.get("termination_exception_send") == "true":
443                        raise
444                    return False
445
446            self._logger.spam("send() succeeded")
447            self._logger.spam("SpeakStreamClient.send LEAVE")
448            return True
449
450        self._logger.spam("send() failed. socket is None")
451        self._logger.spam("SpeakStreamClient.send LEAVE")
452        return False

456    def flush(self) -> bool:
457        """
458        Flushes the current buffer and returns generated audio
459        """
460        self._logger.spam("SpeakStreamClient.flush ENTER")
461
462        if self._exit_event.is_set():
463            self._logger.notice("flush exiting gracefully")
464            self._logger.debug("SpeakStreamClient.flush LEAVE")
465            return False
466
467        if self._socket is None:
468            self._logger.notice("socket is not intialized")
469            self._logger.debug("SpeakStreamClient.flush LEAVE")
470            return False
471
472        self._logger.notice("Sending Flush...")
473        ret = self.send(json.dumps({"type": "Flush"}))
474
475        if not ret:
476            self._logger.error("flush failed")
477            self._logger.spam("SpeakStreamClient.flush LEAVE")
478            return False
479
480        self._logger.notice("flush succeeded")
481        self._logger.spam("SpeakStreamClient.flush LEAVE")
482
483        return True

517    def _signal_exit(self) -> None:
518        # closes the WebSocket connection gracefully
519        self._logger.notice("closing socket...")
520        if self._socket is not None:
521            self._logger.notice("sending Close...")
522            try:
523                # if the socket connection is closed, the following line might throw an error
524                self._socket.send(json.dumps({"type": "Close"}))
525            except websockets.exceptions.ConnectionClosedOK as e:
526                self._logger.notice("_signal_exit  - ConnectionClosedOK: %s", e.code)
527            except websockets.exceptions.ConnectionClosed as e:
528                self._logger.error("_signal_exit  - ConnectionClosed: %s", e.code)
529            except websockets.exceptions.WebSocketException as e:
530                self._logger.error("_signal_exit - WebSocketException: %s", str(e))
531            except Exception as e:  # pylint: disable=broad-except
532                self._logger.error("_signal_exit - Exception: %s", str(e))
533
534            # push close event
535            try:
536                self._emit(
537                    SpeakWebSocketEvents(SpeakWebSocketEvents.Close),
538                    CloseResponse(type=SpeakWebSocketEvents.Close),
539                )
540            except Exception as e:  # pylint: disable=broad-except
541                self._logger.error("_signal_exit - Exception: %s", e)
542
543            # wait for task to send
544            time.sleep(0.5)
545
546        # signal exit
547        self._exit_event.set()
548
549        # closes the WebSocket connection gracefully
550        self._logger.verbose("clean up socket...")
551        if self._socket is not None:
552            self._logger.verbose("socket.wait_closed...")
553            try:
554                self._socket.close()
555            except websockets.exceptions.WebSocketException as e:
556                self._logger.error("socket.wait_closed failed: %s", e)
557
558        self._socket = None  # type: ignore

280                                message
281                            )

286                                **dict(cast(Dict[Any, Any], self._kwargs)),

291                            )

294                                SpeakWebSocketEvents(SpeakWebSocketEvents.Warning),

297                            )
