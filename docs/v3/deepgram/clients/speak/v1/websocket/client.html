149        # add "members" as members of the class

162        elif options is not None:
163            self._options = options
164        else:
165            self._options = {}

167        try:

169            if self._speaker is not None:

255    def _process_text(self, message: str) -> None:
256        """
257        Processes messages received over the WebSocket connection.
258        """
259        self._logger.debug("SpeakWebSocketClient._process_text ENTER")
260
261        try:
262            self._logger.debug("Text data received")
263
264            if len(message) == 0:
265                self._logger.debug("message is empty")
266                self._logger.debug("SpeakWebSocketClient._process_text LEAVE")
267                return
268
269            data = json.loads(message)
270            response_type = data.get("type")
271            self._logger.debug("response_type: %s, data: %s", response_type, data)
272
273            match response_type:
274                case SpeakWebSocketEvents.Open:
275                    open_result: OpenResponse = OpenResponse.from_json(message)
276                    self._logger.verbose("OpenResponse: %s", open_result)
277                    self._emit(
278                        SpeakWebSocketEvents(SpeakWebSocketEvents.Open),
279                        open=open_result,
280                        **dict(cast(Dict[Any, Any], self._kwargs)),
281                    )
282                case SpeakWebSocketEvents.Metadata:
283                    meta_result: MetadataResponse = MetadataResponse.from_json(message)
284                    self._logger.verbose("MetadataResponse: %s", meta_result)
285                    self._emit(
286                        SpeakWebSocketEvents(SpeakWebSocketEvents.Metadata),
287                        metadata=meta_result,
288                        **dict(cast(Dict[Any, Any], self._kwargs)),
289                    )
290                case SpeakWebSocketEvents.Flushed:
291                    fl_result: FlushedResponse = FlushedResponse.from_json(message)
292                    self._logger.verbose("FlushedResponse: %s", fl_result)
293
294                    # auto flush
295                    if self._config.is_inspecting_speak():
296                        with self._lock_flush:
297                            self._flush_count -= 1
298                            self._logger.debug(
299                                "Decrement Flush count: %d",
300                                self._flush_count,
301                            )
302
303                    self._emit(
304                        SpeakWebSocketEvents(SpeakWebSocketEvents.Flushed),
305                        flushed=fl_result,
306                        **dict(cast(Dict[Any, Any], self._kwargs)),
307                    )
308                case SpeakWebSocketEvents.Cleared:
309                    clear_result: ClearedResponse = ClearedResponse.from_json(message)
310                    self._logger.verbose("ClearedResponse: %s", meta_result)
311                    self._emit(
312                        SpeakWebSocketEvents(SpeakWebSocketEvents.Cleared),
313                        cleared=clear_result,
314                        **dict(cast(Dict[Any, Any], self._kwargs)),
315                    )
316                case SpeakWebSocketEvents.Close:
317                    close_result: CloseResponse = CloseResponse.from_json(message)
318                    self._logger.verbose("CloseResponse: %s", close_result)
319                    self._emit(
320                        SpeakWebSocketEvents(SpeakWebSocketEvents.Close),
321                        close=close_result,
322                        **dict(cast(Dict[Any, Any], self._kwargs)),
323                    )
324                case SpeakWebSocketEvents.Warning:
325                    war_warning: WarningResponse = WarningResponse.from_json(message)
326                    self._logger.verbose("WarningResponse: %s", war_warning)
327                    self._emit(
328                        SpeakWebSocketEvents(SpeakWebSocketEvents.Warning),
329                        warning=war_warning,
330                        **dict(cast(Dict[Any, Any], self._kwargs)),
331                    )
332                case SpeakWebSocketEvents.Error:
333                    err_error: ErrorResponse = ErrorResponse.from_json(message)
334                    self._logger.verbose("ErrorResponse: %s", err_error)
335                    self._emit(
336                        SpeakWebSocketEvents(SpeakWebSocketEvents.Error),
337                        error=err_error,
338                        **dict(cast(Dict[Any, Any], self._kwargs)),
339                    )
340                case _:
341                    self._logger.warning(
342                        "Unknown Message: response_type: %s, data: %s",
343                        response_type,
344                        data,
345                    )
346                    unhandled_error: UnhandledResponse = UnhandledResponse(
347                        type=SpeakWebSocketEvents(SpeakWebSocketEvents.Unhandled),
348                        raw=message,
349                    )
350                    self._emit(
351                        SpeakWebSocketEvents(SpeakWebSocketEvents.Unhandled),
352                        unhandled=unhandled_error,
353                        **dict(cast(Dict[Any, Any], self._kwargs)),
354                    )
355
356            self._logger.notice("_process_text Succeeded")
357            self._logger.debug("SpeakWebSocketClient._process_text LEAVE")
358
359        except Exception as e:  # pylint: disable=broad-except
360            self._logger.error("Exception in SpeakWebSocketClient._process_text: %s", e)
361            e_error: ErrorResponse = ErrorResponse(
362                "Exception in SpeakWebSocketClient._process_text",
363                f"{e}",
364                "Exception",
365            )
366            self._logger.error(
367                "Exception in SpeakWebSocketClient._process_text: %s", str(e)
368            )
369            self._emit(
370                SpeakWebSocketEvents(SpeakWebSocketEvents.Error),
371                e_error,
372                **dict(cast(Dict[Any, Any], self._kwargs)),
373            )
374
375            # signal exit and close
376            super()._signal_exit()
377
378            self._logger.debug("SpeakWebSocketClient._process_text LEAVE")
379
380            if self._config.options.get("termination_exception") is True:
381                raise
382            return

400    def _flush(self) -> None:
401        self._logger.debug("SpeakWebSocketClient._flush ENTER")
402
403        delta_in_ms_str = self._config.options.get("auto_flush_speak_delta")
404        if delta_in_ms_str is None:
405            self._logger.error("auto_flush_speak_delta is None")
406            self._logger.debug("SpeakWebSocketClient._flush LEAVE")
407            return
408        delta_in_ms = float(delta_in_ms_str)
409
410        while True:
411            try:
412                self._exit_event.wait(timeout=HALF_SECOND)
413
414                if self._exit_event.is_set():
415                    self._logger.notice("_flush exiting gracefully")
416                    self._logger.debug("ListenWebSocketClient._flush LEAVE")
417                    return
418
419                if self._last_datagram is None:
420                    self._logger.debug("AutoFlush last_datagram is None")
421                    continue
422
423                with self._lock_flush:
424                    delta = datetime.now() - self._last_datagram
425                    diff_in_ms = delta.total_seconds() * 1000
426                    self._logger.debug("AutoFlush delta: %f", diff_in_ms)
427                    if diff_in_ms < delta_in_ms:
428                        self._logger.debug("AutoFlush delta is less than threshold")
429                        continue
430
431                self.flush()
432
433            except Exception as e:  # pylint: disable=broad-except
434                self._logger.error("Exception in SpeakWebSocketClient._flush: %s", e)
435                e_error: ErrorResponse = ErrorResponse(
436                    "Exception in SpeakWebSocketClient._flush",
437                    f"{e}",
438                    "Exception",
439                )
440                self._logger.error(
441                    "Exception in SpeakWebSocketClient._flush: %s", str(e)
442                )
443                self._emit(
444                    SpeakWebSocketEvents(SpeakWebSocketEvents.Error),
445                    error=e_error,
446                    **dict(cast(Dict[Any, Any], self._kwargs)),
447                )
448
449                # signal exit and close
450                super()._signal_exit()
451
452                self._logger.debug("SpeakWebSocketClient._flush LEAVE")
453
454                if self._config.options.get("termination_exception") is True:
455                    raise
456                return

503    def send_raw(self, msg: str) -> bool:
504        """
505        Sends a raw/control message over the WebSocket connection. This message must contain a valid JSON object.
506
507        Args:
508            msg (str): The raw message to send over the WebSocket connection.
509
510        Returns:
511            bool: True if the message was successfully sent, False otherwise.
512        """
513        self._logger.spam("SpeakWebSocketClient.send_raw ENTER")
514
515        if self._config.is_inspecting_speak():
516            try:
517                _tmp_json = json.loads(msg)
518                if "type" in _tmp_json:
519                    self._logger.debug(
520                        "Inspecting Message: Sending %s", _tmp_json["type"]
521                    )
522                    match _tmp_json["type"]:
523                        case SpeakWebSocketMessage.Speak:
524                            inspect_res = self._inspect()
525                            if not inspect_res:
526                                self._logger.error("inspect_res failed")
527                        case SpeakWebSocketMessage.Flush:
528                            with self._lock_flush:
529                                self._last_datagram = None
530                                self._flush_count += 1
531                                self._logger.debug(
532                                    "Increment Flush count: %d", self._flush_count
533                                )
534            except Exception as e:  # pylint: disable=broad-except
535                self._logger.error("send_raw() failed - Exception: %s", str(e))
536
537        try:
538            if super().send(msg) is False:
539                self._logger.error("send_raw() failed")
540                self._logger.spam("SpeakWebSocketClient.send_raw LEAVE")
541                return False
542            self._logger.spam("send_raw() succeeded")
543            self._logger.spam("SpeakWebSocketClient.send_raw LEAVE")
544            return True
545        except Exception as e:  # pylint: disable=broad-except
546            self._logger.error("send_raw() failed - Exception: %s", str(e))
547            self._logger.spam("SpeakWebSocketClient.send_raw LEAVE")
548            if self._config.options.get("termination_exception_send") is True:
549                raise

404        delta_in_ms_str = self._config.options.get("auto_flush_speak_delta")

408            return

411        while True:
412            try:
413                self._exit_event.wait(timeout=HALF_SECOND)

416                    self._logger.notice("_flush exiting gracefully")
417                    self._logger.debug("ListenWebSocketClient._flush LEAVE")
418                    return

420                if self._last_datagram is None:
421                    self._logger.debug("AutoFlush last_datagram is None")
422                    continue
423
424                with self._lock_flush:
425                    delta = datetime.now() - self._last_datagram
426                    diff_in_ms = delta.total_seconds() * 1000
427                    self._logger.debug("AutoFlush delta: %f", diff_in_ms)
428                    if diff_in_ms < delta_in_ms:
429                        self._logger.debug("AutoFlush delta is less than threshold")
430                        continue
431
432                self.flush()
433
434            except Exception as e:  # pylint: disable=broad-except
435                self._logger.error("Exception in SpeakWebSocketClient._flush: %s", e)
