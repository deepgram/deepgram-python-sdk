131        self,
132        options: Optional[Union[SpeakWSOptions, Dict]] = None,
133        addons: Optional[Dict] = None,
134        headers: Optional[Dict] = None,
135        members: Optional[Dict] = None,

138        """
139        Starts the WebSocket connection for text-to-speech synthesis.
140        """
141        self._logger.debug("SpeakWebSocketClient.start ENTER")
142        self._logger.info("options: %s", options)
143        self._logger.info("addons: %s", addons)
144        self._logger.info("headers: %s", headers)
145        self._logger.info("members: %s", members)
146        self._logger.info("kwargs: %s", kwargs)
147
148        if isinstance(options, SpeakWSOptions) and not options.check():
149            self._logger.error("options.check failed")
150            self._logger.debug("SpeakWebSocketClient.start LEAVE")
151            raise DeepgramError("Fatal text-to-speech options error")
152
153        self._addons = addons
154        self._headers = headers
155
156        # add "members" as members of the class
157        if members is not None:
158            self.__dict__.update(members)
159
160        # set kwargs as members of the class
161        if kwargs is not None:
162            self._kwargs = kwargs
163        else:
164            self._kwargs = {}
165
166        if isinstance(options, SpeakWSOptions):
167            self._logger.info("SpeakWSOptions switching class -> dict")
168            self._options = options.to_dict()
169        elif options is not None:
170            self._options = options
171        else:
172            self._options = {}
173
174        try:
175            # speaker substitutes the listening thread
176            if self._speaker is not None:
177                self._logger.notice("passing speaker to delegate_listening")
178                super().delegate_listening(self._speaker)
179
180            # call parent start
181            if (
182                super().start(
183                    self._options,
184                    self._addons,
185                    self._headers,
186                    **dict(cast(Dict[Any, Any], self._kwargs)),
187                )
188                is False
189            ):
190                self._logger.error("SpeakWebSocketClient.start failed")
191                self._logger.debug("SpeakWebSocketClient.start LEAVE")
192                return False
193
194            if self._speaker is not None:
195                self._logger.notice("start delegate_listening thread")
196                self._speaker.start()
197
198            # debug the threads
199            for thread in threading.enumerate():
200                self._logger.debug("after running thread: %s", thread.name)
201            self._logger.debug("number of active threads: %s", threading.active_count())
202
203            # flush thread
204            if self._config.is_auto_flush_speak_enabled():
205                self._logger.notice("autoflush is enabled")
206                self._flush_thread = threading.Thread(target=self._flush)
207                self._flush_thread.start()
208            else:
209                self._logger.notice("autoflush is disabled")
210
211            # debug the threads
212            for thread in threading.enumerate():
213                self._logger.debug("after running thread: %s", thread.name)
214            self._logger.debug("number of active threads: %s", threading.active_count())
215
216            self._logger.notice("start succeeded")
217            self._logger.debug("SpeakWebSocketClient.start LEAVE")

221            self._logger.error(
222                "WebSocketException in SpeakWebSocketClient.start: %s", e

262    def _process_text(self, message: str) -> None:
263        """
264        Processes messages received over the WebSocket connection.
265        """
266        self._logger.debug("SpeakWebSocketClient._process_text ENTER")
267
268        try:
269            self._logger.debug("Text data received")
270
271            if len(message) == 0:
272                self._logger.debug("message is empty")
273                self._logger.debug("SpeakWebSocketClient._process_text LEAVE")
274                return
275
276            data = json.loads(message)
277            response_type = data.get("type")
278            self._logger.debug("response_type: %s, data: %s", response_type, data)
279
280            match response_type:
281                case SpeakWebSocketEvents.Open:
282                    open_result: OpenResponse = OpenResponse.from_json(message)
283                    self._logger.verbose("OpenResponse: %s", open_result)
284                    self._emit(
285                        SpeakWebSocketEvents(SpeakWebSocketEvents.Open),
286                        open=open_result,
287                        **dict(cast(Dict[Any, Any], self._kwargs)),
288                    )
289                case SpeakWebSocketEvents.Metadata:
290                    meta_result: MetadataResponse = MetadataResponse.from_json(message)
291                    self._logger.verbose("MetadataResponse: %s", meta_result)
292                    self._emit(
293                        SpeakWebSocketEvents(SpeakWebSocketEvents.Metadata),
294                        metadata=meta_result,
295                        **dict(cast(Dict[Any, Any], self._kwargs)),
296                    )
297                case SpeakWebSocketEvents.Flushed:
298                    fl_result: FlushedResponse = FlushedResponse.from_json(message)
299                    self._logger.verbose("FlushedResponse: %s", fl_result)
300
301                    # auto flush
302                    if self._config.is_inspecting_speak():
303                        with self._lock_flush:
304                            self._flush_count -= 1
305                            self._logger.debug(
306                                "Decrement Flush count: %d",
307                                self._flush_count,
308                            )
309
310                    self._emit(
311                        SpeakWebSocketEvents(SpeakWebSocketEvents.Flushed),
312                        flushed=fl_result,
313                        **dict(cast(Dict[Any, Any], self._kwargs)),
314                    )
315                case SpeakWebSocketEvents.Cleared:
316                    clear_result: ClearedResponse = ClearedResponse.from_json(message)
317                    self._logger.verbose("ClearedResponse: %s", meta_result)
318                    self._emit(
319                        SpeakWebSocketEvents(SpeakWebSocketEvents.Cleared),
320                        cleared=clear_result,
321                        **dict(cast(Dict[Any, Any], self._kwargs)),
322                    )
323                case SpeakWebSocketEvents.Close:
324                    close_result: CloseResponse = CloseResponse.from_json(message)
325                    self._logger.verbose("CloseResponse: %s", close_result)
326                    self._emit(
327                        SpeakWebSocketEvents(SpeakWebSocketEvents.Close),
328                        close=close_result,
329                        **dict(cast(Dict[Any, Any], self._kwargs)),
330                    )
331                case SpeakWebSocketEvents.Warning:
332                    war_warning: WarningResponse = WarningResponse.from_json(message)
333                    self._logger.verbose("WarningResponse: %s", war_warning)
334                    self._emit(
335                        SpeakWebSocketEvents(SpeakWebSocketEvents.Warning),
336                        warning=war_warning,
337                        **dict(cast(Dict[Any, Any], self._kwargs)),
338                    )
339                case SpeakWebSocketEvents.Error:
340                    err_error: ErrorResponse = ErrorResponse.from_json(message)
341                    self._logger.verbose("ErrorResponse: %s", err_error)
342                    self._emit(
343                        SpeakWebSocketEvents(SpeakWebSocketEvents.Error),
344                        error=err_error,
345                        **dict(cast(Dict[Any, Any], self._kwargs)),
346                    )
347                case _:
348                    self._logger.warning(
349                        "Unknown Message: response_type: %s, data: %s",
350                        response_type,
351                        data,
352                    )
353                    unhandled_error: UnhandledResponse = UnhandledResponse(
354                        type=SpeakWebSocketEvents(SpeakWebSocketEvents.Unhandled),
355                        raw=message,
356                    )
357                    self._emit(
358                        SpeakWebSocketEvents(SpeakWebSocketEvents.Unhandled),
359                        unhandled=unhandled_error,
360                        **dict(cast(Dict[Any, Any], self._kwargs)),
361                    )
362
363            self._logger.notice("_process_text Succeeded")
364            self._logger.debug("SpeakWebSocketClient._process_text LEAVE")
365
366        except Exception as e:  # pylint: disable=broad-except
367            self._logger.error("Exception in SpeakWebSocketClient._process_text: %s", e)
368            e_error: ErrorResponse = ErrorResponse(
369                "Exception in SpeakWebSocketClient._process_text",
370                f"{e}",
371                "Exception",
372            )
373            self._logger.error(
374                "Exception in SpeakWebSocketClient._process_text: %s", str(e)
375            )
376            self._emit(
377                SpeakWebSocketEvents(SpeakWebSocketEvents.Error),
378                e_error,
379                **dict(cast(Dict[Any, Any], self._kwargs)),
380            )
381
382            # signal exit and close
383            super()._signal_exit()
384
385            self._logger.debug("SpeakWebSocketClient._process_text LEAVE")
386
387            if self._config.options.get("termination_exception") is True:
388                raise
389            return

407    def _flush(self) -> None:
408        self._logger.debug("SpeakWebSocketClient._flush ENTER")
409
410        delta_in_ms_str = self._config.options.get("auto_flush_speak_delta")
411        if delta_in_ms_str is None:
412            self._logger.error("auto_flush_speak_delta is None")
413            self._logger.debug("SpeakWebSocketClient._flush LEAVE")
414            return
415        delta_in_ms = float(delta_in_ms_str)
416
417        while True:
418            try:
419                self._exit_event.wait(timeout=HALF_SECOND)
420
421                if self._exit_event.is_set():
422                    self._logger.notice("_flush exiting gracefully")
423                    self._logger.debug("ListenWebSocketClient._flush LEAVE")
424                    return
425
426                if self._last_datagram is None:
427                    self._logger.debug("AutoFlush last_datagram is None")
428                    continue
429
430                with self._lock_flush:
431                    delta = datetime.now() - self._last_datagram
432                    diff_in_ms = delta.total_seconds() * 1000
433                    self._logger.debug("AutoFlush delta: %f", diff_in_ms)
434                    if diff_in_ms < delta_in_ms:
435                        self._logger.debug("AutoFlush delta is less than threshold")
436                        continue
437
438                self.flush()
439
440            except Exception as e:  # pylint: disable=broad-except
441                self._logger.error("Exception in SpeakWebSocketClient._flush: %s", e)
442                e_error: ErrorResponse = ErrorResponse(
443                    "Exception in SpeakWebSocketClient._flush",
444                    f"{e}",
445                    "Exception",
446                )
447                self._logger.error(
448                    "Exception in SpeakWebSocketClient._flush: %s", str(e)
449                )
450                self._emit(
451                    SpeakWebSocketEvents(SpeakWebSocketEvents.Error),
452                    error=e_error,
453                    **dict(cast(Dict[Any, Any], self._kwargs)),
454                )
455
456                # signal exit and close
457                super()._signal_exit()
458
459                self._logger.debug("SpeakWebSocketClient._flush LEAVE")
460
461                if self._config.options.get("termination_exception") is True:
462                    raise
463                return

510    def send_raw(self, msg: str) -> bool:
511        """
512        Sends a raw/control message over the WebSocket connection. This message must contain a valid JSON object.
513
514        Args:
515            msg (str): The raw message to send over the WebSocket connection.
516
517        Returns:
518            bool: True if the message was successfully sent, False otherwise.
519        """
520        self._logger.spam("SpeakWebSocketClient.send_raw ENTER")
521
522        if self._config.is_inspecting_speak():
523            try:
524                _tmp_json = json.loads(msg)
525                if "type" in _tmp_json:
526                    self._logger.debug(
527                        "Inspecting Message: Sending %s", _tmp_json["type"]
528                    )
529                    match _tmp_json["type"]:
530                        case SpeakWebSocketMessage.Speak:
531                            inspect_res = self._inspect()
532                            if not inspect_res:
533                                self._logger.error("inspect_res failed")
534                        case SpeakWebSocketMessage.Flush:
535                            with self._lock_flush:
536                                self._last_datagram = None
537                                self._flush_count += 1
538                                self._logger.debug(
539                                    "Increment Flush count: %d", self._flush_count
540                                )
541            except Exception as e:  # pylint: disable=broad-except
542                self._logger.error("send_raw() failed - Exception: %s", str(e))
543
544        try:
545            if super().send(msg) is False:
546                self._logger.error("send_raw() failed")
547                self._logger.spam("SpeakWebSocketClient.send_raw LEAVE")
548                return False
549            self._logger.spam("send_raw() succeeded")
550            self._logger.spam("SpeakWebSocketClient.send_raw LEAVE")
551            return True
552        except Exception as e:  # pylint: disable=broad-except
553            self._logger.error("send_raw() failed - Exception: %s", str(e))
554            self._logger.spam("SpeakWebSocketClient.send_raw LEAVE")
555            if self._config.options.get("termination_exception_send") is True:
556                raise
557            return False
