 91    async def start(
 92        self,
 93        options: Optional[Union[ListenWebSocketOptions, Dict]] = None,
 94        addons: Optional[Dict] = None,
 95        headers: Optional[Dict] = None,
 96        members: Optional[Dict] = None,
 97        **kwargs,
 98    ) -> bool:
 99        """
100        Starts the WebSocket connection for live transcription.
101        """
102        self._logger.debug("AsyncListenWebSocketClient.start ENTER")
103        self._logger.info("options: %s", options)
104        self._logger.info("addons: %s", addons)
105        self._logger.info("headers: %s", headers)
106        self._logger.info("members: %s", members)
107        self._logger.info("kwargs: %s", kwargs)
108
109        if isinstance(options, ListenWebSocketOptions) and not options.check():
110            self._logger.error("options.check failed")
111            self._logger.debug("AsyncListenWebSocketClient.start LEAVE")
112            raise DeepgramError("Fatal transcription options error")
113
114        self._addons = addons
115        self._headers = headers
116
117        # add "members" as members of the class
118        if members is not None:
119            self.__dict__.update(members)
120
121        # set kwargs as members of the class
122        if kwargs is not None:
123            self._kwargs = kwargs
124        else:
125            self._kwargs = {}
126
127        if isinstance(options, ListenWebSocketOptions):
128            self._logger.info("ListenWebSocketOptions switching class -> dict")
129            self._options = options.to_dict()
130        elif options is not None:
131            self._options = options
132        else:
133            self._options = {}
134
135        combined_options = self._options
136        if self._addons is not None:
137            self._logger.info("merging addons to options")
138            combined_options.update(self._addons)
139            self._logger.info("new options: %s", combined_options)
140        self._logger.debug("combined_options: %s", combined_options)
141
142        combined_headers = self._config.headers
143        if self._headers is not None:
144            self._logger.info("merging headers to options")
145            combined_headers.update(self._headers)
146            self._logger.info("new headers: %s", combined_headers)
147        self._logger.debug("combined_headers: %s", combined_headers)
148
149        url_with_params = append_query_params(self._websocket_url, combined_options)
150
151        try:
152            self._socket = await websockets.connect(
153                url_with_params,
154                extra_headers=combined_headers,
155                ping_interval=PING_INTERVAL,
156            )
157            self._exit_event.clear()
158
159            # debug the threads
160            for thread in threading.enumerate():
161                self._logger.debug("after running thread: %s", thread.name)
162            self._logger.debug("number of active threads: %s", threading.active_count())
163
164            # listen thread
165            self._listen_thread = asyncio.create_task(self._listening())
166
167            # keepalive thread
168            if self._config.is_keep_alive_enabled():
169                self._logger.notice("keepalive is enabled")
170                self._keep_alive_thread = asyncio.create_task(self._keep_alive())
171            else:
172                self._logger.notice("keepalive is disabled")
173
174            # flush thread
175            if self._config.is_auto_flush_reply_enabled():
176                self._logger.notice("autoflush is enabled")
177                self._flush_thread = asyncio.create_task(self._flush())
178            else:
179                self._logger.notice("autoflush is disabled")
180
181            # debug the threads
182            for thread in threading.enumerate():
183                self._logger.debug("after running thread: %s", thread.name)
184            self._logger.debug("number of active threads: %s", threading.active_count())
185
186            # push open event
187            await self._emit(
188                LiveTranscriptionEvents(LiveTranscriptionEvents.Open),
189                OpenResponse(type=LiveTranscriptionEvents.Open),
190            )
191
192            self._logger.notice("start succeeded")
193            self._logger.debug("AsyncListenWebSocketClient.start LEAVE")
194            return True
195        except websockets.ConnectionClosed as e:
196            self._logger.error(
197                "ConnectionClosed in AsyncListenWebSocketClient.start: %s", e
198            )
199            self._logger.debug("AsyncListenWebSocketClient.start LEAVE")
200            if self._config.options.get("termination_exception_connect") == "true":
201                raise
202            return False
203        except websockets.exceptions.WebSocketException as e:
204            self._logger.error(
205                "WebSocketException in AsyncListenWebSocketClient.start: %s", e
206            )
207            self._logger.debug("AsyncListenWebSocketClient.start LEAVE")
208            if self._config.options.get("termination_exception_connect") == "true":
209                raise
210            return False
211        except Exception as e:  # pylint: disable=broad-except
212            self._logger.error(
213                "WebSocketException in AsyncListenWebSocketClient.start: %s", e
214            )
215            self._logger.debug("AsyncListenWebSocketClient.start LEAVE")
216            if self._config.options.get("termination_exception_connect") == "true":
217                raise
218            return False

228    def on(self, event: LiveTranscriptionEvents, handler) -> None:
229        """
230        Registers event handlers for specific events.
231        """
232        self._logger.info("event subscribed: %s", event)
233        if event in LiveTranscriptionEvents.__members__.values() and callable(handler):
234            if handler not in self._event_handlers[event]:
235                self._event_handlers[event].append(handler)

238    async def _emit(self, event: LiveTranscriptionEvents, *args, **kwargs) -> None:
239        """
240        Emits events to the registered event handlers.
241        """
242        self._logger.debug("AsyncListenWebSocketClient._emit ENTER")
243        self._logger.debug("callback handlers for: %s", event)
244
245        # debug the threads
246        for thread in threading.enumerate():
247            self._logger.debug("after running thread: %s", thread.name)
248        self._logger.debug("number of active threads: %s", threading.active_count())
249
250        tasks = []
251        for handler in self._event_handlers[event]:
252            task = asyncio.create_task(handler(self, *args, **kwargs))
253            tasks.append(task)
254
255        if tasks:
256            self._logger.debug("waiting for tasks to finish...")
257            await asyncio.gather(*tasks, return_exceptions=True)
258            tasks.clear()
259
260        # debug the threads
261        for thread in threading.enumerate():
262            self._logger.debug("after running thread: %s", thread.name)
263        self._logger.debug("number of active threads: %s", threading.active_count())
264
265        self._logger.debug("AsyncListenWebSocketClient._emit LEAVE")

268    async def _listening(self) -> None:
269        """
270        Listens for messages from the WebSocket connection.
271        """
272        self._logger.debug("AsyncListenWebSocketClient._listening ENTER")
273
274        while True:
275            try:
276                if self._exit_event.is_set():
277                    self._logger.notice("_listening exiting gracefully")
278                    self._logger.debug("AsyncListenWebSocketClient._listening LEAVE")
279                    return
280
281                if self._socket is None:
282                    self._logger.warning("socket is empty")
283                    self._logger.debug("AsyncListenWebSocketClient._listening LEAVE")
284                    return
285
286                message = str(await self._socket.recv())
287
288                if message is None:
289                    self._logger.spam("message is None")
290                    continue
291
292                data = json.loads(message)
293                response_type = data.get("type")
294                self._logger.debug("response_type: %s, data: %s", response_type, data)
295
296                match response_type:
297                    case LiveTranscriptionEvents.Open:
298                        open_result: OpenResponse = OpenResponse.from_json(message)
299                        self._logger.verbose("OpenResponse: %s", open_result)
300                        await self._emit(
301                            LiveTranscriptionEvents(LiveTranscriptionEvents.Open),
302                            open=open_result,
303                            **dict(cast(Dict[Any, Any], self._kwargs)),
304                        )
305                    case LiveTranscriptionEvents.Transcript:
306                        msg_result: LiveResultResponse = LiveResultResponse.from_json(
307                            message
308                        )
309                        self._logger.verbose("LiveResultResponse: %s", msg_result)
310
311                        # auto flush
312                        if self._config.is_inspecting_listen():
313                            inspect_res = await self._inspect(msg_result)
314                            if not inspect_res:
315                                self._logger.error("inspect_res failed")
316
317                        await self._emit(
318                            LiveTranscriptionEvents(LiveTranscriptionEvents.Transcript),
319                            result=msg_result,
320                            **dict(cast(Dict[Any, Any], self._kwargs)),
321                        )
322                    case LiveTranscriptionEvents.Metadata:
323                        meta_result: MetadataResponse = MetadataResponse.from_json(
324                            message
325                        )
326                        self._logger.verbose("MetadataResponse: %s", meta_result)
327                        await self._emit(
328                            LiveTranscriptionEvents(LiveTranscriptionEvents.Metadata),
329                            metadata=meta_result,
330                            **dict(cast(Dict[Any, Any], self._kwargs)),
331                        )
332                    case LiveTranscriptionEvents.SpeechStarted:
333                        ss_result: SpeechStartedResponse = (
334                            SpeechStartedResponse.from_json(message)
335                        )
336                        self._logger.verbose("SpeechStartedResponse: %s", ss_result)
337                        await self._emit(
338                            LiveTranscriptionEvents(
339                                LiveTranscriptionEvents.SpeechStarted
340                            ),
341                            speech_started=ss_result,
342                            **dict(cast(Dict[Any, Any], self._kwargs)),
343                        )
344                    case LiveTranscriptionEvents.UtteranceEnd:
345                        ue_result: UtteranceEndResponse = (
346                            UtteranceEndResponse.from_json(message)
347                        )
348                        self._logger.verbose("UtteranceEndResponse: %s", ue_result)
349                        await self._emit(
350                            LiveTranscriptionEvents(
351                                LiveTranscriptionEvents.UtteranceEnd
352                            ),
353                            utterance_end=ue_result,
354                            **dict(cast(Dict[Any, Any], self._kwargs)),
355                        )
356                    case LiveTranscriptionEvents.Close:
357                        close_result: CloseResponse = CloseResponse.from_json(message)
358                        self._logger.verbose("CloseResponse: %s", close_result)
359                        await self._emit(
360                            LiveTranscriptionEvents(LiveTranscriptionEvents.Close),
361                            close=close_result,
362                            **dict(cast(Dict[Any, Any], self._kwargs)),
363                        )
364                    case LiveTranscriptionEvents.Error:
365                        err_error: ErrorResponse = ErrorResponse.from_json(message)
366                        self._logger.verbose("ErrorResponse: %s", err_error)
367                        await self._emit(
368                            LiveTranscriptionEvents(LiveTranscriptionEvents.Error),
369                            error=err_error,
370                            **dict(cast(Dict[Any, Any], self._kwargs)),
371                        )
372                    case _:
373                        self._logger.warning(
374                            "Unknown Message: response_type: %s, data: %s",
375                            response_type,
376                            data,
377                        )
378                        unhandled_error: UnhandledResponse = UnhandledResponse(
379                            type=LiveTranscriptionEvents(
380                                LiveTranscriptionEvents.Unhandled
381                            ),
382                            raw=message,
383                        )
384                        await self._emit(
385                            LiveTranscriptionEvents(LiveTranscriptionEvents.Unhandled),
386                            unhandled=unhandled_error,
387                            **dict(cast(Dict[Any, Any], self._kwargs)),
388                        )
389
390            except websockets.exceptions.ConnectionClosedOK as e:
391                self._logger.notice(f"_listening({e.code}) exiting gracefully")
392                self._logger.debug("AsyncListenWebSocketClient._listening LEAVE")
393                return
394
395            except websockets.exceptions.ConnectionClosed as e:
396                if e.code in [1000, 1001]:
397                    self._logger.notice(f"_listening({e.code}) exiting gracefully")
398                    self._logger.debug("AsyncListenWebSocketClient._listening LEAVE")
399                    return
400
401                # we need to explicitly call self._signal_exit() here because we are hanging on a recv()
402                # note: this is different than the speak websocket client
403                self._logger.error(
404                    "ConnectionClosed in AsyncListenWebSocketClient._listening with code %s: %s",
405                    e.code,
406                    e.reason,
407                )
408                cc_error: ErrorResponse = ErrorResponse(
409                    "ConnectionClosed in AsyncListenWebSocketClient._listening",
410                    f"{e}",
411                    "ConnectionClosed",
412                )
413                await self._emit(
414                    LiveTranscriptionEvents(LiveTranscriptionEvents.Error),
415                    error=cc_error,
416                    **dict(cast(Dict[Any, Any], self._kwargs)),
417                )
418
419                # signal exit and close
420                await self._signal_exit()
421
422                self._logger.debug("AsyncListenWebSocketClient._listening LEAVE")
423
424                if self._config.options.get("termination_exception") == "true":
425                    raise
426                return
427
428            except websockets.exceptions.WebSocketException as e:
429                self._logger.error(
430                    "WebSocketException in AsyncListenWebSocketClient._listening: %s", e
431                )
432                ws_error: ErrorResponse = ErrorResponse(
433                    "WebSocketException in AsyncListenWebSocketClient._listening",
434                    f"{e}",
435                    "WebSocketException",
436                )
437                await self._emit(
438                    LiveTranscriptionEvents(LiveTranscriptionEvents.Error),
439                    error=ws_error,
440                    **dict(cast(Dict[Any, Any], self._kwargs)),
441                )
442
443                # signal exit and close
444                await self._signal_exit()
445
446                self._logger.debug("AsyncListenWebSocketClient._listening LEAVE")
447
448                if self._config.options.get("termination_exception") == "true":
449                    raise
450                return
451
452            except Exception as e:  # pylint: disable=broad-except
453                self._logger.error(
454                    "Exception in AsyncListenWebSocketClient._listening: %s", e
455                )
456                e_error: ErrorResponse = ErrorResponse(
457                    "Exception in AsyncListenWebSocketClient._listening",
458                    f"{e}",
459                    "Exception",
460                )
461                await self._emit(
462                    LiveTranscriptionEvents(LiveTranscriptionEvents.Error),
463                    error=e_error,
464                    **dict(cast(Dict[Any, Any], self._kwargs)),
465                )
466
467                # signal exit and close
468                await self._signal_exit()
469
470                self._logger.debug("AsyncListenWebSocketClient._listening LEAVE")
471
472                if self._config.options.get("termination_exception") == "true":
473                    raise
474                return

596    async def _flush(self) -> None:
597        self._logger.debug("AsyncListenWebSocketClient._flush ENTER")
598
599        delta_in_ms_str = self._config.options.get("auto_flush_reply_delta")
600        if delta_in_ms_str is None:
601            self._logger.error("auto_flush_reply_delta is None")
602            self._logger.debug("AsyncListenWebSocketClient._flush LEAVE")
603            return
604        delta_in_ms = float(delta_in_ms_str)
605
606        while True:
607            try:
608                await asyncio.sleep(HALF_SECOND)
609
610                if self._exit_event.is_set():
611                    self._logger.notice("_flush exiting gracefully")
612                    self._logger.debug("AsyncListenWebSocketClient._flush LEAVE")
613                    return
614
615                if self._socket is None:
616                    self._logger.notice("socket is None, exiting flush")
617                    self._logger.debug("AsyncListenWebSocketClient._flush LEAVE")
618                    return
619
620                if self._last_datagram is None:
621                    self._logger.debug("AutoFlush last_datagram is None")
622                    continue
623
624                delta = datetime.now() - self._last_datagram
625                diff_in_ms = delta.total_seconds() * 1000
626                self._logger.debug("AutoFlush delta: %f", diff_in_ms)
627                if diff_in_ms < delta_in_ms:
628                    self._logger.debug("AutoFlush delta is less than threshold")
629                    continue
630
631                self._last_datagram = None
632                await self.finalize()
633
634            except websockets.exceptions.ConnectionClosedOK as e:
635                self._logger.notice(f"_flush({e.code}) exiting gracefully")
636                self._logger.debug("AsyncListenWebSocketClient._flush LEAVE")
637                return
638
639            except websockets.exceptions.ConnectionClosed as e:
640                if e.code in [1000, 1001]:
641                    self._logger.notice(f"_flush({e.code}) exiting gracefully")
642                    self._logger.debug("AsyncListenWebSocketClient._flush LEAVE")
643                    return
644
645                # we need to explicitly call self._signal_exit() here because we are hanging on a recv()
646                # note: this is different than the speak websocket client
647                self._logger.error(
648                    "ConnectionClosed in AsyncListenWebSocketClient._flush with code %s: %s",
649                    e.code,
650                    e.reason,
651                )
652                cc_error: ErrorResponse = ErrorResponse(
653                    "ConnectionClosed in AsyncListenWebSocketClient._flush",
654                    f"{e}",
655                    "ConnectionClosed",
656                )
657                await self._emit(
658                    LiveTranscriptionEvents(LiveTranscriptionEvents.Error),
659                    error=cc_error,
660                    **dict(cast(Dict[Any, Any], self._kwargs)),
661                )
662
663                # signal exit and close
664                await self._signal_exit()
665
666                self._logger.debug("AsyncListenWebSocketClient._flush LEAVE")
667
668                if self._config.options.get("termination_exception") == "true":
669                    raise
670                return
671
672            except websockets.exceptions.WebSocketException as e:
673                self._logger.error(
674                    "WebSocketException in AsyncListenWebSocketClient._flush: %s", e
675                )
676                ws_error: ErrorResponse = ErrorResponse(
677                    "WebSocketException in AsyncListenWebSocketClient._flush",
678                    f"{e}",
679                    "Exception",
680                )
681                await self._emit(
