225    async def _process_text(self, message: str) -> None:
226        """
227        Processes messages received over the WebSocket connection.
228        """
229        self._logger.debug("AsyncListenWebSocketClient._process_text ENTER")
230
231        try:
232            self._logger.debug("Text data received")
233            if len(message) == 0:
234                self._logger.debug("message is empty")
235                self._logger.debug("AsyncListenWebSocketClient._process_text LEAVE")
236                return
237
238            data = json.loads(message)
239            response_type = data.get("type")
240            self._logger.debug("response_type: %s, data: %s", response_type, data)
241
242            match response_type:
243                case LiveTranscriptionEvents.Open:
244                    open_result: OpenResponse = OpenResponse.from_json(message)
245                    self._logger.verbose("OpenResponse: %s", open_result)
246                    await self._emit(
247                        LiveTranscriptionEvents(LiveTranscriptionEvents.Open),
248                        open=open_result,
249                        **dict(cast(Dict[Any, Any], self._kwargs)),
250                    )
251                case LiveTranscriptionEvents.Transcript:
252                    msg_result: LiveResultResponse = LiveResultResponse.from_json(
253                        message
254                    )
255                    self._logger.verbose("LiveResultResponse: %s", msg_result)
256
257                    # auto flush
258                    if self._config.is_inspecting_listen():
259                        inspect_res = await self._inspect(msg_result)
260                        if not inspect_res:
261                            self._logger.error("inspect_res failed")
262
263                    await self._emit(
264                        LiveTranscriptionEvents(LiveTranscriptionEvents.Transcript),
265                        result=msg_result,
266                        **dict(cast(Dict[Any, Any], self._kwargs)),
267                    )
268                case LiveTranscriptionEvents.Metadata:
269                    meta_result: MetadataResponse = MetadataResponse.from_json(message)
270                    self._logger.verbose("MetadataResponse: %s", meta_result)
271                    await self._emit(
272                        LiveTranscriptionEvents(LiveTranscriptionEvents.Metadata),
273                        metadata=meta_result,
274                        **dict(cast(Dict[Any, Any], self._kwargs)),
275                    )
276                case LiveTranscriptionEvents.SpeechStarted:
277                    ss_result: SpeechStartedResponse = SpeechStartedResponse.from_json(
278                        message
279                    )
280                    self._logger.verbose("SpeechStartedResponse: %s", ss_result)
281                    await self._emit(
282                        LiveTranscriptionEvents(LiveTranscriptionEvents.SpeechStarted),
283                        speech_started=ss_result,
284                        **dict(cast(Dict[Any, Any], self._kwargs)),
285                    )
286                case LiveTranscriptionEvents.UtteranceEnd:
287                    ue_result: UtteranceEndResponse = UtteranceEndResponse.from_json(
288                        message
289                    )
290                    self._logger.verbose("UtteranceEndResponse: %s", ue_result)
291                    await self._emit(
292                        LiveTranscriptionEvents(LiveTranscriptionEvents.UtteranceEnd),
293                        utterance_end=ue_result,
294                        **dict(cast(Dict[Any, Any], self._kwargs)),
295                    )
296                case LiveTranscriptionEvents.Close:
297                    close_result: CloseResponse = CloseResponse.from_json(message)
298                    self._logger.verbose("CloseResponse: %s", close_result)
299                    await self._emit(
300                        LiveTranscriptionEvents(LiveTranscriptionEvents.Close),
301                        close=close_result,
302                        **dict(cast(Dict[Any, Any], self._kwargs)),
303                    )
304                case LiveTranscriptionEvents.Error:
305                    err_error: ErrorResponse = ErrorResponse.from_json(message)
306                    self._logger.verbose("ErrorResponse: %s", err_error)
307                    await self._emit(
308                        LiveTranscriptionEvents(LiveTranscriptionEvents.Error),
309                        error=err_error,
310                        **dict(cast(Dict[Any, Any], self._kwargs)),
311                    )
312                case _:
313                    self._logger.warning(
314                        "Unknown Message: response_type: %s, data: %s",
315                        response_type,
316                        data,
317                    )
318                    unhandled_error: UnhandledResponse = UnhandledResponse(
319                        type=LiveTranscriptionEvents(LiveTranscriptionEvents.Unhandled),
320                        raw=message,
321                    )
322                    await self._emit(
323                        LiveTranscriptionEvents(LiveTranscriptionEvents.Unhandled),
324                        unhandled=unhandled_error,
325                        **dict(cast(Dict[Any, Any], self._kwargs)),
326                    )
327
328            self._logger.notice("_process_text Succeeded")
329            self._logger.debug("AsyncListenWebSocketClient._process_text LEAVE")
330
331        except Exception as e:  # pylint: disable=broad-except
332            self._logger.error(
333                "Exception in AsyncListenWebSocketClient._process_text: %s", e
334            )
335            e_error: ErrorResponse = ErrorResponse(
336                "Exception in AsyncListenWebSocketClient._process_text",
337                f"{e}",
338                "Exception",
339            )
340            await self._emit(
341                LiveTranscriptionEvents(LiveTranscriptionEvents.Error),
342                error=e_error,
343                **dict(cast(Dict[Any, Any], self._kwargs)),
344            )
345
346            # signal exit and close
347            await super()._signal_exit()
348
349            self._logger.debug("AsyncListenWebSocketClient._process_text LEAVE")
350
351            if self._config.options.get("termination_exception") is True:
352                raise
353            return

361    async def _keep_alive(self) -> None:
362        """
363        Sends keepalive messages to the WebSocket connection.
364        """
365        self._logger.debug("AsyncListenWebSocketClient._keep_alive ENTER")
366
367        counter = 0
368        while True:
369            try:
370                counter += 1
371                await asyncio.sleep(ONE_SECOND)
372
373                if self._exit_event.is_set():
374                    self._logger.notice("_keep_alive exiting gracefully")
375                    self._logger.debug("AsyncListenWebSocketClient._keep_alive LEAVE")
376                    return
377
378                # deepgram keepalive
379                if counter % DEEPGRAM_INTERVAL == 0:
380                    await self.keep_alive()
381
382            except Exception as e:  # pylint: disable=broad-except
383                self._logger.error(
384                    "Exception in AsyncListenWebSocketClient._keep_alive: %s", e
385                )
386                e_error: ErrorResponse = ErrorResponse(
387                    "Exception in AsyncListenWebSocketClient._keep_alive",
388                    f"{e}",
389                    "Exception",
390                )
391                self._logger.error(
392                    "Exception in AsyncListenWebSocketClient._keep_alive: %s", str(e)
393                )
394                await self._emit(
395                    LiveTranscriptionEvents(LiveTranscriptionEvents.Error),
396                    error=e_error,
397                    **dict(cast(Dict[Any, Any], self._kwargs)),
398                )
399
400                # signal exit and close
401                await super()._signal_exit()
402
403                self._logger.debug("AsyncListenWebSocketClient._keep_alive LEAVE")

410    async def _flush(self) -> None:
411        self._logger.debug("AsyncListenWebSocketClient._flush ENTER")
412
413        delta_in_ms_str = self._config.options.get("auto_flush_reply_delta")
414        if delta_in_ms_str is None:
415            self._logger.error("auto_flush_reply_delta is None")
416            self._logger.debug("AsyncListenWebSocketClient._flush LEAVE")
417            return
418        delta_in_ms = float(delta_in_ms_str)
419
420        while True:
421            try:
422                await asyncio.sleep(HALF_SECOND)
423
424                if self._exit_event.is_set():
425                    self._logger.notice("_flush exiting gracefully")
426                    self._logger.debug("AsyncListenWebSocketClient._flush LEAVE")
427                    return
428
429                if self._last_datagram is None:
430                    self._logger.debug("AutoFlush last_datagram is None")
431                    continue
432
433                delta = datetime.now() - self._last_datagram
434                diff_in_ms = delta.total_seconds() * 1000
435                self._logger.debug("AutoFlush delta: %f", diff_in_ms)
436                if diff_in_ms < delta_in_ms:
437                    self._logger.debug("AutoFlush delta is less than threshold")
438                    continue
439
440                self._last_datagram = None
441                await self.finalize()
442
443            except Exception as e:  # pylint: disable=broad-except
444                self._logger.error(
445                    "Exception in AsyncListenWebSocketClient._flush: %s", e
446                )
447                e_error: ErrorResponse = ErrorResponse(
448                    "Exception in AsyncListenWebSocketClient._flush",
449                    f"{e}",
450                    "Exception",
451                )
452                self._logger.error(
453                    "Exception in AsyncListenWebSocketClient._flush: %s", str(e)
454                )
455                await self._emit(
456                    LiveTranscriptionEvents(LiveTranscriptionEvents.Error),
457                    error=e_error,
458                    **dict(cast(Dict[Any, Any], self._kwargs)),
459                )
460
461                # signal exit and close
462                await super()._signal_exit()
463
464                self._logger.debug("AsyncListenWebSocketClient._flush LEAVE")
465
466                if self._config.options.get("termination_exception") is True:
467                    raise
468                return

516        """
517        self._logger.debug("AsyncListenWebSocketClient.finish ENTER")
518

527            # debug the threads
528            for thread in threading.enumerate():

559            self._logger.debug("AsyncListenWebSocketClient.finish LEAVE")
