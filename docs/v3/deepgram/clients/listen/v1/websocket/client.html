224    def _process_text(self, message: str) -> None:
225        """
226        Processes messages received over the WebSocket connection.
227        """
228        self._logger.debug("ListenWebSocketClient._process_text ENTER")
229
230        try:
231            if len(message) == 0:
232                self._logger.debug("message is empty")
233                self._logger.debug("ListenWebSocketClient._process_text LEAVE")
234                return
235
236            data = json.loads(message)
237            response_type = data.get("type")
238            self._logger.debug("response_type: %s, data: %s", response_type, data)
239
240            match response_type:
241                case LiveTranscriptionEvents.Open:
242                    open_result: OpenResponse = OpenResponse.from_json(message)
243                    self._logger.verbose("OpenResponse: %s", open_result)
244                    self._emit(
245                        LiveTranscriptionEvents(LiveTranscriptionEvents.Open),
246                        open=open_result,
247                        **dict(cast(Dict[Any, Any], self._kwargs)),
248                    )
249                case LiveTranscriptionEvents.Transcript:
250                    msg_result: LiveResultResponse = LiveResultResponse.from_json(
251                        message
252                    )
253                    self._logger.verbose("LiveResultResponse: %s", msg_result)
254
255                    #  auto flush
256                    if self._config.is_inspecting_listen():
257                        inspect_res = self._inspect(msg_result)
258                        if not inspect_res:
259                            self._logger.error("inspect_res failed")
260
261                    self._emit(
262                        LiveTranscriptionEvents(LiveTranscriptionEvents.Transcript),
263                        result=msg_result,
264                        **dict(cast(Dict[Any, Any], self._kwargs)),
265                    )
266                case LiveTranscriptionEvents.Metadata:
267                    meta_result: MetadataResponse = MetadataResponse.from_json(message)
268                    self._logger.verbose("MetadataResponse: %s", meta_result)
269                    self._emit(
270                        LiveTranscriptionEvents(LiveTranscriptionEvents.Metadata),
271                        metadata=meta_result,
272                        **dict(cast(Dict[Any, Any], self._kwargs)),
273                    )
274                case LiveTranscriptionEvents.SpeechStarted:
275                    ss_result: SpeechStartedResponse = SpeechStartedResponse.from_json(
276                        message
277                    )
278                    self._logger.verbose("SpeechStartedResponse: %s", ss_result)
279                    self._emit(
280                        LiveTranscriptionEvents(LiveTranscriptionEvents.SpeechStarted),
281                        speech_started=ss_result,
282                        **dict(cast(Dict[Any, Any], self._kwargs)),
283                    )
284                case LiveTranscriptionEvents.UtteranceEnd:
285                    ue_result: UtteranceEndResponse = UtteranceEndResponse.from_json(
286                        message
287                    )
288                    self._logger.verbose("UtteranceEndResponse: %s", ue_result)
289                    self._emit(
290                        LiveTranscriptionEvents(LiveTranscriptionEvents.UtteranceEnd),
291                        utterance_end=ue_result,
292                        **dict(cast(Dict[Any, Any], self._kwargs)),
293                    )
294                case LiveTranscriptionEvents.Close:
295                    close_result: CloseResponse = CloseResponse.from_json(message)
296                    self._logger.verbose("CloseResponse: %s", close_result)
297                    self._emit(
298                        LiveTranscriptionEvents(LiveTranscriptionEvents.Close),
299                        close=close_result,
300                        **dict(cast(Dict[Any, Any], self._kwargs)),
301                    )
302                case LiveTranscriptionEvents.Error:
303                    err_error: ErrorResponse = ErrorResponse.from_json(message)
304                    self._logger.verbose("ErrorResponse: %s", err_error)
305                    self._emit(
306                        LiveTranscriptionEvents(LiveTranscriptionEvents.Error),
307                        error=err_error,
308                        **dict(cast(Dict[Any, Any], self._kwargs)),
309                    )
310                case _:
311                    self._logger.warning(
312                        "Unknown Message: response_type: %s, data: %s",
313                        response_type,
314                        data,
315                    )
316                    unhandled_error: UnhandledResponse = UnhandledResponse(
317                        type=LiveTranscriptionEvents(LiveTranscriptionEvents.Unhandled),
318                        raw=message,
319                    )
320                    self._emit(
321                        LiveTranscriptionEvents(LiveTranscriptionEvents.Unhandled),
322                        unhandled=unhandled_error,
323                        **dict(cast(Dict[Any, Any], self._kwargs)),
324                    )
325
326            self._logger.notice("_process_text Succeeded")
327            self._logger.debug("SpeakStreamClient._process_text LEAVE")
328
329        except Exception as e:  # pylint: disable=broad-except
330            self._logger.error(
331                "Exception in ListenWebSocketClient._process_text: %s", e
332            )
333            e_error: ErrorResponse = ErrorResponse(
334                "Exception in ListenWebSocketClient._process_text",
335                f"{e}",
336                "Exception",
337            )
338            self._logger.error(
339                "Exception in ListenWebSocketClient._process_text: %s", str(e)
340            )
341            self._emit(
342                LiveTranscriptionEvents(LiveTranscriptionEvents.Error),
343                e_error,
344                **dict(cast(Dict[Any, Any], self._kwargs)),
345            )
346
347            # signal exit and close
348            super()._signal_exit()
349
350            self._logger.debug("ListenWebSocketClient._process_text LEAVE")
351
352            if self._config.options.get("termination_exception") is True:
353                raise
354            return

362    def _keep_alive(self) -> None:
363        self._logger.debug("ListenWebSocketClient._keep_alive ENTER")
364
365        counter = 0
366        while True:

408    def _flush(self) -> None:
409        self._logger.debug("ListenWebSocketClient._flush ENTER")
410
411        delta_in_ms_str = self._config.options.get("auto_flush_reply_delta")
412        if delta_in_ms_str is None:
413            self._logger.error("auto_flush_reply_delta is None")
414            self._logger.debug("ListenWebSocketClient._flush LEAVE")
415            return
416        delta_in_ms = float(delta_in_ms_str)
417
418        _flush_event = threading.Event()
419        while True:
420            try:
421                _flush_event.wait(timeout=HALF_SECOND)
422
423                if self._exit_event.is_set():
424                    self._logger.notice("_flush exiting gracefully")
425                    self._logger.debug("ListenWebSocketClient._flush LEAVE")
426                    return
427
428                with self._lock_flush:
429                    if self._last_datagram is None:
430                        self._logger.debug("AutoFlush last_datagram is None")
431                        continue
432
433                    delta = datetime.now() - self._last_datagram
434                    diff_in_ms = delta.total_seconds() * 1000
435                    self._logger.debug("AutoFlush delta: %f", diff_in_ms)
436                    if diff_in_ms < delta_in_ms:
437                        self._logger.debug("AutoFlush delta is less than threshold")
438                        continue
439
440                    with self._lock_flush:
441                        self._last_datagram = None
442                    self.finalize()
443
444            except Exception as e:  # pylint: disable=broad-except
445                self._logger.error("Exception in ListenWebSocketClient._flush: %s", e)
446                e_error: ErrorResponse = ErrorResponse(
447                    "Exception in ListenWebSocketClient._flush",
448                    f"{e}",
449                    "Exception",
450                )
451                self._logger.error(
452                    "Exception in ListenWebSocketClient._flush: %s", str(e)
453                )
454                self._emit(
455                    LiveTranscriptionEvents(LiveTranscriptionEvents.Error),
456                    e_error,
457                    **dict(cast(Dict[Any, Any], self._kwargs)),
458                )
459
460                # signal exit and close
461                super()._signal_exit()
462
463                self._logger.debug("ListenWebSocketClient._flush LEAVE")
464
465                if self._config.options.get("termination_exception") is True:
466                    raise
467                return

554    def _inspect(self, msg_result: LiveResultResponse) -> bool:
555        # auto flush_inspect is generically used to track any messages you might want to snoop on
556        # place additional logic here to inspect messages of interest
557
558        # for auto flush functionality
559        # set the last datagram
560        sentence = msg_result.channel.alternatives[0].transcript
561        if len(sentence) == 0:
562            return True
563
564        if msg_result.is_final:
565            with self._lock_flush:
566                self._logger.debug("AutoFlush is_final received")
567                self._last_datagram = None
568        else:
569            with self._lock_flush:
570                self._last_datagram = datetime.now()
571                self._logger.debug(
572                    "AutoFlush interim received: %s",
573                    str(self._last_datagram),
574                )
575
576        return True
