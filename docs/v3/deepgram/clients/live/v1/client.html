100    def start(
101        self,
102        options: Optional[Union[LiveOptions, Dict]] = None,
103        addons: Optional[Dict] = None,
104        headers: Optional[Dict] = None,
105        members: Optional[Dict] = None,
106        **kwargs,
107    ) -> bool:
108        """
109        Starts the WebSocket connection for live transcription.
110        """
111        self._logger.debug("LiveClient.start ENTER")
112        self._logger.info("options: %s", options)
113        self._logger.info("addons: %s", addons)
114        self._logger.info("headers: %s", headers)
115        self._logger.info("members: %s", members)
116        self._logger.info("kwargs: %s", kwargs)
117
118        if isinstance(options, LiveOptions) and not options.check():
119            self._logger.error("options.check failed")
120            self._logger.debug("LiveClient.start LEAVE")
121            raise DeepgramError("Fatal transcription options error")
122
123        self._addons = addons
124        self._headers = headers
125
126        # add "members" as members of the class
127        if members is not None:
128            self.__dict__.update(members)
129
130        # set kwargs as members of the class
131        if kwargs is not None:
132            self._kwargs = kwargs
133        else:
134            self._kwargs = {}
135
136        if isinstance(options, LiveOptions):
137            self._logger.info("LiveOptions switching class -> dict")
138            self._options = options.to_dict()
139        elif options is not None:
140            self._options = options
141        else:
142            self._options = {}
143
144        combined_options = self._options
145        if self._addons is not None:
146            self._logger.info("merging addons to options")
147            combined_options.update(self._addons)
148            self._logger.info("new options: %s", combined_options)
149        self._logger.debug("combined_options: %s", combined_options)
150
151        combined_headers = self._config.headers
152        if self._headers is not None:
153            self._logger.info("merging headers to options")
154            combined_headers.update(self._headers)
155            self._logger.info("new headers: %s", combined_headers)
156        self._logger.debug("combined_headers: %s", combined_headers)
157
158        url_with_params = append_query_params(self._websocket_url, combined_options)
159        try:
160            self._socket = connect(url_with_params, additional_headers=combined_headers)
161            self._exit_event.clear()
162
163            # debug the threads
164            for thread in threading.enumerate():
165                self._logger.debug("after running thread: %s", thread.name)
166            self._logger.debug("number of active threads: %s", threading.active_count())
167
168            # listening thread
169            self._listen_thread = threading.Thread(target=self._listening)
170            self._listen_thread.start()
171
172            # keepalive thread
173            if self._config.is_keep_alive_enabled():
174                self._logger.notice("keepalive is enabled")
175                self._keep_alive_thread = threading.Thread(target=self._keep_alive)
176                self._keep_alive_thread.start()
177            else:
178                self._logger.notice("keepalive is disabled")
179
180            # flush thread
181            if self._config.is_auto_flush_enabled():
182                self._logger.notice("autoflush is enabled")
183                self._flush_thread = threading.Thread(target=self._flush)
184                self._flush_thread.start()
185            else:
186                self._logger.notice("autoflush is disabled")
187
188            # debug the threads
189            for thread in threading.enumerate():
190                self._logger.debug("after running thread: %s", thread.name)
191            self._logger.debug("number of active threads: %s", threading.active_count())
192
193            # push open event
194            self._emit(
195                LiveTranscriptionEvents(LiveTranscriptionEvents.Open),
196                OpenResponse(type=LiveTranscriptionEvents.Open),
197            )
198
199            self._logger.notice("start succeeded")
200            self._logger.debug("LiveClient.start LEAVE")
201            return True
202        except websockets.ConnectionClosed as e:
203            self._logger.error("ConnectionClosed in LiveClient.start: %s", e)
204            self._logger.debug("LiveClient.start LEAVE")
205            if self._config.options.get("termination_exception_connect") == "true":
206                raise e
207            return False
208        except websockets.exceptions.WebSocketException as e:
209            self._logger.error("WebSocketException in LiveClient.start: %s", e)
210            self._logger.debug("LiveClient.start LEAVE")
211            if self._config.options.get("termination_exception_connect") == "true":
212                raise e
213            return False
214        except Exception as e:  # pylint: disable=broad-except
215            self._logger.error("WebSocketException in LiveClient.start: %s", e)
216            self._logger.debug("LiveClient.start LEAVE")
217            if self._config.options.get("termination_exception_connect") == "true":
218                raise e
219            return False

223    def on(
224        self, event: LiveTranscriptionEvents, handler
225    ) -> None:  # registers event handlers for specific events
226        """
227        Registers event handlers for specific events.
228        """
229        self._logger.info("event subscribed: %s", event)
230        if event in LiveTranscriptionEvents.__members__.values() and callable(handler):
231            self._event_handlers[event].append(handler)

233    def _emit(self, event: LiveTranscriptionEvents, *args, **kwargs) -> None:
234        """
235        Emits events to the registered event handlers.
236        """
237        self._logger.debug("callback handlers for: %s", event)
238        for handler in self._event_handlers[event]:
239            handler(self, *args, **kwargs)

242    def _listening(
243        self,
244    ) -> None:
245        """
246        Listens for messages from the WebSocket connection.
247        """
248        self._logger.debug("LiveClient._listening ENTER")
249
250        while True:
251            try:
252                if self._exit_event.is_set():
253                    self._logger.notice("_listening exiting gracefully")
254                    self._logger.debug("LiveClient._listening LEAVE")
255                    return
256
257                if self._socket is None:
258                    self._logger.warning("socket is empty")
259                    self._logger.debug("LiveClient._listening LEAVE")
260                    return
261
262                message = str(self._socket.recv())
263
264                if message is None:
265                    self._logger.info("message is empty")
266                    continue
267
268                data = json.loads(message)
269                response_type = data.get("type")
270                self._logger.debug("response_type: %s, data: %s", response_type, data)
271
272                match response_type:
273                    case LiveTranscriptionEvents.Open:
274                        open_result: OpenResponse = OpenResponse.from_json(message)
275                        self._logger.verbose("OpenResponse: %s", open_result)
276                        self._emit(
277                            LiveTranscriptionEvents(LiveTranscriptionEvents.Open),
278                            open=open_result,
279                            **dict(cast(Dict[Any, Any], self._kwargs)),
280                        )
281                    case LiveTranscriptionEvents.Transcript:
282                        msg_result: LiveResultResponse = LiveResultResponse.from_json(
283                            message
284                        )
285                        self._logger.verbose("LiveResultResponse: %s", msg_result)
286
287                        #  auto flush
288                        if self._config.is_inspecting_messages():
289                            inspect_res = self._inspect(msg_result)
290                            if not inspect_res:
291                                self._logger.error("inspect_res failed")
292
293                        self._emit(
294                            LiveTranscriptionEvents(LiveTranscriptionEvents.Transcript),
295                            result=msg_result,
296                            **dict(cast(Dict[Any, Any], self._kwargs)),
297                        )
298                    case LiveTranscriptionEvents.Metadata:
299                        meta_result: MetadataResponse = MetadataResponse.from_json(
300                            message
301                        )
302                        self._logger.verbose("MetadataResponse: %s", meta_result)
303                        self._emit(
304                            LiveTranscriptionEvents(LiveTranscriptionEvents.Metadata),
305                            metadata=meta_result,
306                            **dict(cast(Dict[Any, Any], self._kwargs)),
307                        )
308                    case LiveTranscriptionEvents.SpeechStarted:
309                        ss_result: SpeechStartedResponse = (
310                            SpeechStartedResponse.from_json(message)
311                        )
312                        self._logger.verbose("SpeechStartedResponse: %s", ss_result)
313                        self._emit(
314                            LiveTranscriptionEvents(
315                                LiveTranscriptionEvents.SpeechStarted
316                            ),
317                            speech_started=ss_result,
318                            **dict(cast(Dict[Any, Any], self._kwargs)),
319                        )
320                    case LiveTranscriptionEvents.UtteranceEnd:
321                        ue_result: UtteranceEndResponse = (
322                            UtteranceEndResponse.from_json(message)
323                        )
324                        self._logger.verbose("UtteranceEndResponse: %s", ue_result)
325                        self._emit(
326                            LiveTranscriptionEvents(
327                                LiveTranscriptionEvents.UtteranceEnd
328                            ),
329                            utterance_end=ue_result,
330                            **dict(cast(Dict[Any, Any], self._kwargs)),
331                        )
332                    case LiveTranscriptionEvents.Close:
333                        close_result: CloseResponse = CloseResponse.from_json(message)
334                        self._logger.verbose("CloseResponse: %s", close_result)
335                        self._emit(
336                            LiveTranscriptionEvents(LiveTranscriptionEvents.Close),
337                            close=close_result,
338                            **dict(cast(Dict[Any, Any], self._kwargs)),
339                        )
340                    case LiveTranscriptionEvents.Error:
341                        err_error: ErrorResponse = ErrorResponse.from_json(message)
342                        self._logger.verbose("ErrorResponse: %s", err_error)
343                        self._emit(
344                            LiveTranscriptionEvents(LiveTranscriptionEvents.Error),
345                            error=err_error,
346                            **dict(cast(Dict[Any, Any], self._kwargs)),
347                        )
348                    case _:
349                        self._logger.warning(
350                            "Unknown Message: response_type: %s, data: %s",
351                            response_type,
352                            data,
353                        )
354                        unhandled_error: UnhandledResponse = UnhandledResponse(
355                            type=LiveTranscriptionEvents(
356                                LiveTranscriptionEvents.Unhandled
357                            ),
358                            raw=message,
359                        )
360                        self._emit(
361                            LiveTranscriptionEvents(LiveTranscriptionEvents.Unhandled),
362                            unhandled=unhandled_error,
363                            **dict(cast(Dict[Any, Any], self._kwargs)),
364                        )
365
366            except websockets.exceptions.ConnectionClosedOK as e:
367                self._logger.notice(f"_listening({e.code}) exiting gracefully")
368                self._logger.debug("LiveClient._listening LEAVE")
369                return
370
371            except websockets.exceptions.ConnectionClosed as e:
372                if e.code == 1000:
373                    self._logger.notice(f"_listening({e.code}) exiting gracefully")
374                    self._logger.debug("LiveClient._listening LEAVE")
375                    return
376
377                self._logger.error(
378                    "ConnectionClosed in LiveClient._listening with code %s: %s",
379                    e.code,
380                    e.reason,
381                )
382                cc_error: ErrorResponse = ErrorResponse(
383                    "ConnectionClosed in LiveClient._listening",
384                    f"{e}",
385                    "ConnectionClosed",
386                )
387                self._emit(
388                    LiveTranscriptionEvents(LiveTranscriptionEvents.Error), cc_error
389                )
390
391                # signal exit and close
392                self._signal_exit()
393
394                self._logger.debug("LiveClient._listening LEAVE")
395
396                if self._config.options.get("termination_exception") == "true":
397                    raise
398                return
399
400            except websockets.exceptions.WebSocketException as e:
401                self._logger.error(
402                    "WebSocketException in LiveClient._listening with: %s", e
403                )
404                ws_error: ErrorResponse = ErrorResponse(
405                    "WebSocketException in LiveClient._listening",
406                    f"{e}",
407                    "WebSocketException",
408                )
409                self._emit(
410                    LiveTranscriptionEvents(LiveTranscriptionEvents.Error), ws_error
411                )
412
413                # signal exit and close
414                self._signal_exit()
415
416                self._logger.debug("LiveClient._listening LEAVE")
417
418                if self._config.options.get("termination_exception") == "true":
419                    raise
420                return
421
422            except Exception as e:  # pylint: disable=broad-except
423                self._logger.error("Exception in LiveClient._listening: %s", e)
424                e_error: ErrorResponse = ErrorResponse(
425                    "Exception in LiveClient._listening",
426                    f"{e}",
427                    "Exception",
428                )
429                self._logger.error("Exception in LiveClient._listening: %s", str(e))
430                self._emit(
431                    LiveTranscriptionEvents(LiveTranscriptionEvents.Error), e_error
432                )
433
434                # signal exit and close
435                self._signal_exit()
436
437                self._logger.debug("LiveClient._listening LEAVE")
438
439                if self._config.options.get("termination_exception") == "true":
440                    raise
441                return

549    def _flush(self) -> None:
550        self._logger.debug("LiveClient._flush ENTER")
551
552        delta_in_ms_str = self._config.options.get("auto_flush_reply_delta")
553        if delta_in_ms_str is None:
554            self._logger.error("auto_flush_reply_delta is None")
555            self._logger.debug("LiveClient._flush LEAVE")
556            return
557        delta_in_ms = float(delta_in_ms_str)
558
559        while True:
560            try:
561                self._flush_event.wait(timeout=HALF_SECOND)
562
563                if self._exit_event.is_set():
564                    self._logger.notice("_flush exiting gracefully")
565                    self._logger.debug("LiveClient._flush LEAVE")
566                    return
567
568                if self._socket is None:
569                    self._logger.debug("socket is None, exiting flush")
570                    self._logger.debug("LiveClient._flush LEAVE")
571                    return
572
573                with self._lock_flush:
574                    if self._last_datagram is None:
575                        self._logger.debug("AutoFlush last_datagram is None")
576                        continue
577
578                    delta = datetime.now() - self._last_datagram
579                    diff_in_ms = delta.total_seconds() * 1000
580                    self._logger.debug("AutoFlush delta: %f", diff_in_ms)
581                    if diff_in_ms < delta_in_ms:
582                        self._logger.debug("AutoFlush delta is less than threshold")
583                        continue
584
585                    with self._lock_flush:
586                        self._last_datagram = None
587                    self.finalize()
588
589            except websockets.exceptions.ConnectionClosedOK as e:
590                self._logger.notice(f"_flush({e.code}) exiting gracefully")
591                self._logger.debug("LiveClient._flush LEAVE")
592                return
593
594            except websockets.exceptions.ConnectionClosed as e:
595                if e.code == 1000:
596                    self._logger.notice(f"_flush({e.code}) exiting gracefully")
597                    self._logger.debug("LiveClient._flush LEAVE")
598                    return
599
600                self._logger.error(
601                    "ConnectionClosed in LiveClient._flush with code %s: %s",
602                    e.code,
603                    e.reason,
604                )
605                cc_error: ErrorResponse = ErrorResponse(
606                    "ConnectionClosed in LiveClient._flush",
607                    f"{e}",
608                    "ConnectionClosed",
609                )
610                self._emit(
611                    LiveTranscriptionEvents(LiveTranscriptionEvents.Error), cc_error
612                )
613
614                # signal exit and close
615                self._signal_exit()
616
617                self._logger.debug("LiveClient._flush LEAVE")
618
619                if self._config.options.get("termination_exception") == "true":
620                    raise
621                return
622
623            except websockets.exceptions.WebSocketException as e:
624                self._logger.error(
625                    "WebSocketException in LiveClient._flush with: %s", e
626                )
627                ws_error: ErrorResponse = ErrorResponse(
628                    "WebSocketException in LiveClient._flush",
629                    f"{e}",
630                    "WebSocketException",
631                )
632                self._emit(
633                    LiveTranscriptionEvents(LiveTranscriptionEvents.Error), ws_error
634                )
635
636                # signal exit and close
637                self._signal_exit()
638
639                self._logger.debug("LiveClient._flush LEAVE")
640
641                if self._config.options.get("termination_exception") == "true":
642                    raise
643                return
644
645            except Exception as e:  # pylint: disable=broad-except
646                self._logger.error("Exception in LiveClient._flush: %s", e)
647                e_error: ErrorResponse = ErrorResponse(
648                    "Exception in LiveClient._flush",
649                    f"{e}",

669    def send(self, data: Union[str, bytes]) -> bool:
670        """
671        Sends data over the WebSocket connection.
672        """
673        self._logger.spam("LiveClient.send ENTER")
674
675        if self._exit_event.is_set():
676            self._logger.notice("send exiting gracefully")
677            self._logger.debug("LiveClient.send LEAVE")
678            return False
679
680        if self._socket is not None:
681            with self._lock_send:
682                try:
683                    self._socket.send(data)
684                except websockets.exceptions.ConnectionClosedOK as e:
685                    self._logger.notice(f"send() exiting gracefully: {e.code}")
686                    self._logger.debug("LiveClient._keep_alive LEAVE")
687                    if self._config.options.get("termination_exception_send") == "true":
688                        raise
689                    return True
690                except websockets.exceptions.ConnectionClosed as e:
691                    if e.code == 1000:
692                        self._logger.notice(f"send({e.code}) exiting gracefully")
693                        self._logger.debug("LiveClient.send LEAVE")
694                        if (
695                            self._config.options.get("termination_exception_send")
696                            == "true"
697                        ):
698                            raise
699                        return True
700                    self._logger.error("send() failed - ConnectionClosed: %s", str(e))
701                    self._logger.spam("LiveClient.send LEAVE")
702                    if self._config.options.get("termination_exception_send") == "true":
703                        raise
