 36    Client for interacting with Deepgram's live transcription services over WebSockets.

 44    def __init__(self, config: DeepgramClientOptions):
 45        if config is None:
 46            raise DeepgramError("Config are required")
 47
 48        self.logger = logging.getLogger(__name__)
 49        self.logger.addHandler(logging.StreamHandler())
 50        self.logger.setLevel(config.verbose)
 51
 52        self.config = config
 53        self.endpoint = "v1/listen"
 54        self._socket = None
 55        self._exit_event = threading.Event()
 56        self._lock_send = threading.Lock()
 57        self._event_handlers = {event: [] for event in LiveTranscriptionEvents}
 58        self.websocket_url = convert_to_websocket_url(self.config.url, self.endpoint)

 60    # starts the WebSocket connection for live transcription
 61    def start(
 62        self,
 63        options: Optional[Union[LiveOptions, Dict]] = None,
 64        addons: Optional[Dict] = None,
 65        headers: Optional[Dict] = None,
 66        members: Optional[Dict] = None,
 67        **kwargs,
 68    ) -> bool:
 69        """
 70        Starts the WebSocket connection for live transcription.
 71        """
 72        self.logger.debug("LiveClient.start ENTER")
 73        self.logger.info("options: %s", options)
 74        self.logger.info("addons: %s", addons)
 75        self.logger.info("headers: %s", headers)
 76        self.logger.info("members: %s", members)
 77        self.logger.info("kwargs: %s", kwargs)
 78
 79        if isinstance(options, LiveOptions) and not options.check():
 80            self.logger.error("options.check failed")
 81            self.logger.debug("LiveClient.start LEAVE")
 82            raise DeepgramError("Fatal transcription options error")
 83
 84        if self._socket is not None:
 85            self.logger.error("socket is already initialized")
 86            self.logger.debug("LiveClient.start LEAVE")
 87            raise DeepgramWebsocketError("Websocket already started")
 88
 89        self.options = options
 90        self.addons = addons
 91        self.headers = headers
 92
 93        # add "members" as members of the class
 94        if members is not None:
 95            self.__dict__.update(members)
 96
 97        # set kwargs as members of the class
 98        if kwargs is not None:
 99            self.kwargs = kwargs
100        else:
101            self.kwargs = dict()
102
103        if isinstance(options, LiveOptions):
104            self.logger.info("LiveOptions switching class -> dict")
105            self.options = self.options.to_dict()
106
107        combined_options = self.options
108        if addons is not None:
109            self.logger.info("merging addons to options")
110            combined_options.update(self.addons)
111            self.logger.info("new options: %s", combined_options)
112        self.logger.debug("combined_options: %s", combined_options)
113
114        combined_headers = self.config.headers
115        if headers is not None:
116            self.logger.info("merging headers to options")
117            combined_headers.update(self.headers)
118            self.logger.info("new headers: %s", combined_headers)
119        self.logger.debug("combined_headers: %s", combined_headers)
120
121        url_with_params = append_query_params(self.websocket_url, combined_options)
122        try:
123            self._socket = connect(url_with_params, additional_headers=combined_headers)
124            self._exit_event.clear()
125
126            # listening thread
127            self._listen_thread = threading.Thread(target=self._listening)
128            self._listen_thread.start()
129
130            # keepalive thread
131            if self.config.options.get("keepalive") == "true":
132                self.logger.notice("keepalive is enabled")
133                self._keep_alive_thread = threading.Thread(target=self._keep_alive)
134                self._keep_alive_thread.start()
135            else:
136                self.logger.notice("keepalive is disabled")
137                self._keep_alive_thread = None
138
139            # push open event
140            self._emit(
141                LiveTranscriptionEvents.Open,
142                OpenResponse(type=LiveTranscriptionEvents.Open.value),
143            )
144
145            self.logger.notice("start succeeded")
146            self.logger.debug("LiveClient.start LEAVE")
147            return True
148        except websockets.ConnectionClosed as e:
149            self.logger.error("exception: websockets.ConnectionClosed")
150            self.logger.debug("LiveClient.start LEAVE")
151            if self.config.options.get("termination_exception_connect") == "true":
152                raise
153            return False
154        except websockets.exceptions.WebSocketException as e:
155            self.logger.error("WebSocketException in LiveClient.start: %s", e)
156            self.logger.debug("LiveClient.start LEAVE")
157            if self.config.options.get("termination_exception_connect") == "true":
158                raise
159            return False
160        except Exception as e:
161            self.logger.error("WebSocketException in LiveClient.start: %s", e)
162            self.logger.debug("LiveClient.start LEAVE")
163            if self.config.options.get("termination_exception_connect") == "true":
164                raise
165            return False

178        for handler in self._event_handlers[event]:
179            handler(self, *args, **kwargs)

181    # main loop for handling incoming messages
182    def _listening(self) -> None:
183        self.logger.debug("LiveClient._listening ENTER")
184
185        while True:
186            try:
187                if self._exit_event.is_set():
188                    self.logger.notice("_listening exiting gracefully")
189                    self.logger.debug("LiveClient._listening LEAVE")
190                    return
191
192                if self._socket is None:
193                    self.logger.warning("socket is empty")
194                    self.logger.debug("LiveClient._listening LEAVE")
195                    return
196
197                message = self._socket.recv()
198
199                if message is None:
200                    self.logger.info("message is empty")
201                    continue
202
203                data = json.loads(message)
204                response_type = data.get("type")
205                self.logger.debug("response_type: %s, data: %s", response_type, data)
206
207                match response_type:
208                    case LiveTranscriptionEvents.Open.value:
209                        result = OpenResponse.from_json(message)
210                        self.logger.verbose("OpenResponse: %s", result)
211                        self._emit(
212                            LiveTranscriptionEvents.Open,
213                            open=result,
214                            **dict(self.kwargs),
215                        )
216                    case LiveTranscriptionEvents.Transcript.value:
217                        result = LiveResultResponse.from_json(message)
218                        self.logger.verbose("LiveResultResponse: %s", result)
219                        self._emit(
220                            LiveTranscriptionEvents.Transcript,
221                            result=result,
222                            **dict(self.kwargs),
223                        )
224                    case LiveTranscriptionEvents.Metadata.value:
225                        result = MetadataResponse.from_json(message)
226                        self.logger.verbose("MetadataResponse: %s", result)
227                        self._emit(
228                            LiveTranscriptionEvents.Metadata,
229                            metadata=result,
230                            **dict(self.kwargs),
231                        )
232                    case LiveTranscriptionEvents.SpeechStarted.value:
233                        result = SpeechStartedResponse.from_json(message)
234                        self.logger.verbose("SpeechStartedResponse: %s", result)
235                        self._emit(
236                            LiveTranscriptionEvents.SpeechStarted,
237                            speech_started=result,
238                            **dict(self.kwargs),
239                        )
240                    case LiveTranscriptionEvents.UtteranceEnd.value:
241                        result = UtteranceEndResponse.from_json(message)
242                        self.logger.verbose("UtteranceEndResponse: %s", result)
243                        self._emit(
244                            LiveTranscriptionEvents.UtteranceEnd,
245                            utterance_end=result,
246                            **dict(self.kwargs),
247                        )
248                    case LiveTranscriptionEvents.Close.value:
249                        result = CloseResponse.from_json(message)
250                        self.logger.verbose("CloseResponse: %s", result)
251                        self._emit(
252                            LiveTranscriptionEvents.Close,
253                            close=result,
254                            **dict(self.kwargs),
255                        )
256                    case LiveTranscriptionEvents.Error.value:
257                        result = ErrorResponse.from_json(message)
258                        self.logger.verbose("ErrorResponse: %s", result)
259                        self._emit(
260                            LiveTranscriptionEvents.Error,
261                            error=result,
262                            **dict(self.kwargs),
263                        )
264                    case _:
265                        self.logger.warning(
266                            "Unknown Message: response_type: %s, data: %s",
267                            response_type,
268                            data,
269                        )
270                        unhandled = UnhandledResponse(
271                            type=LiveTranscriptionEvents.Unhandled.value,
272                            raw=message,
273                        )
274                        self._emit(
275                            LiveTranscriptionEvents.Unhandled,
276                            unhandled=unhandled,
277                            **dict(self.kwargs),
278                        )
279
280            except websockets.exceptions.ConnectionClosedOK as e:
281                self.logger.notice(f"_listening({e.code}) exiting gracefully")
282                self.logger.debug("LiveClient._listening LEAVE")
283                return
284
285            except websockets.exceptions.WebSocketException as e:
286                if e.code == 1000:
287                    self.logger.notice(f"_listening({e.code}) exiting gracefully")
288                    self.logger.debug("LiveClient._listening LEAVE")
289                    return
290
291                self.logger.error(
292                    "WebSocketException in AsyncLiveClient._listening: %s", e
293                )
294                error: ErrorResponse = {
295                    "type": "Exception",
296                    "description": "WebSocketException in LiveClient._listening",
297                    "message": f"{e}",
298                    "variant": "",
299                }
300                self.logger.notice(
301                    f"WebSocket connection in LiveClient._listening closed with code {e.code}: {e.reason}"
302                )
303                self._emit(LiveTranscriptionEvents.Error, error)
304
305                # signal exit and close
306                self._signal_exit()
307
308                self.logger.debug("LiveClient._listening LEAVE")
309
310                if self.config.options.get("termination_exception") == "true":
311                    raise
312                return
313
314            except Exception as e:
315                self.logger.error("Exception in AsyncLiveClient._listening: %s", e)
316                error: ErrorResponse = {
317                    "type": "Exception",
318                    "description": "Exception in LiveClient._listening",
319                    "message": f"{e}",
320                    "variant": "",
321                }
322                self.logger.error("Exception in LiveClient._listening: %s", str(e))
323                self._emit(LiveTranscriptionEvents.Error, error)
324
325                # signal exit and close
326                self._signal_exit()
327
328                self.logger.debug("LiveClient._listening LEAVE")
329
330                if self.config.options.get("termination_exception") == "true":
331                    raise
332                return

334    # keep the connection alive by sending keepalive messages
335    def _keep_alive(self) -> None:
336        self.logger.debug("LiveClient._keep_alive ENTER")
337
338        counter = 0
339        while True:
340            try:
341                counter += 1
342
343                self._exit_event.wait(timeout=ONE_SECOND)
344                if self._exit_event.is_set():
345                    self.logger.notice("_keep_alive exiting gracefully")
346                    self.logger.debug("LiveClient._keep_alive LEAVE")
347                    return
348
349                if self._socket is None:
350                    self.logger.notice("socket is None, exiting keep_alive")
351                    self.logger.debug("LiveClient._keep_alive LEAVE")
352                    return
353
354                # deepgram keepalive
355                if counter % DEEPGRAM_INTERVAL == 0:
356                    self.logger.verbose("Sending KeepAlive...")
357                    self.send(json.dumps({"type": "KeepAlive"}))
358
359            except websockets.exceptions.ConnectionClosedOK as e:
360                self.logger.notice(f"_keep_alive({e.code}) exiting gracefully")
361                self.logger.debug("LiveClient._keep_alive LEAVE")
362                return
363
364            except websockets.exceptions.WebSocketException as e:
365                if e.code == 1000:
366                    self.logger.notice(f"_keep_alive({e.code}) exiting gracefully")
367                    self.logger.debug("LiveClient._keep_alive LEAVE")
368                    return
369
370                self.logger.error(
371                    "WebSocketException in AsyncLiveClient._keep_alive: %s", e
372                )
373                error: ErrorResponse = {
374                    "type": "Exception",
375                    "description": "WebSocketException in LiveClient._keep_alive",
376                    "message": f"{e}",
377                    "variant": "",
378                }
379                self.logger.error(

383
384                # signal exit and close

392
393            except Exception as e:

395                error: ErrorResponse = {
396                    "type": "Exception",
397                    "description": "Exception in LiveClient._keep_alive",

402                self._emit(LiveTranscriptionEvents.Error, error)
403
404                # signal exit and close
405                self._signal_exit()
406
407                self.logger.debug("LiveClient._keep_alive LEAVE")

409                if self.config.options.get("termination_exception") == "true":

413    # sends data over the WebSocket connection
414    def send(self, data: Union[str, bytes]) -> bool:
415        """
416        Sends data over the WebSocket connection.
417        """
418        self.logger.spam("LiveClient.send ENTER")
419
420        if self._exit_event.is_set():
421            self.logger.notice("send exiting gracefully")
422            self.logger.debug("AsyncLiveClient.send LEAVE")
423            return False
424
425        if self._socket is not None:
426            with self._lock_send:
427                try:
428                    self._socket.send(data)
429                except websockets.exceptions.ConnectionClosedOK as e:
430                    self.logger.notice(f"send() exiting gracefully: {e.code}")
431                    self.logger.debug("LiveClient._keep_alive LEAVE")
432                    if self.config.options.get("termination_exception_send") == "true":
433                        raise
434                    return True
435                except websockets.exceptions.WebSocketException as e:
436                    if e.code == 1000:
437                        self.logger.notice(f"send({e.code}) exiting gracefully")
438                        self.logger.debug("LiveClient.send LEAVE")
439                        if (
440                            self.config.options.get("termination_exception_send")
441                            == "true"
442                        ):
443                            raise
444                        return True
445                    self.logger.error("send() failed - WebSocketException: %s", str(e))
446                    self.logger.spam("LiveClient.send LEAVE")
447                    if self.config.options.get("termination_exception_send") == "true":
448                        raise
449                    return False
450                except Exception as e:
451                    self.logger.error("send() failed - Exception: %s", str(e))
452                    self.logger.spam("LiveClient.send LEAVE")
453                    if self.config.options.get("termination_exception_send") == "true":
454                        raise
455                    return False
456
457            self.logger.spam(f"send() succeeded")
458            self.logger.spam("LiveClient.send LEAVE")
459            return True
460
461        self.logger.spam("send() failed. socket is None")
462        self.logger.spam("LiveClient.send LEAVE")
463        return False

465    # closes the WebSocket connection gracefully
466    def finish(self) -> bool:
467        """
468        Closes the WebSocket connection gracefully.
469        """
470        self.logger.spam("LiveClient.finish ENTER")
471
472        # signal exit
473        self._signal_exit()
474
475        # stop the threads
476        self.logger.verbose("cancelling tasks...")
477        if self._keep_alive_thread is not None:
478            self._keep_alive_thread.join()
479        self.logger.notice("processing thread joined")
480
481        if self._listen_thread is not None:
482            self._listen_thread.join()
483        self.logger.notice("listening thread joined")
484
485        self._socket = None
486
487        self.logger.notice("finish succeeded")
488        self.logger.spam("LiveClient.finish LEAVE")
489        return True

491    # signals the WebSocket connection to exit
492    def _signal_exit(self) -> None:
493        # closes the WebSocket connection gracefully
494        self.logger.notice("closing socket...")
495        if self._socket is not None:
496            self.logger.notice("sending CloseStream...")
497            try:
498                # if the socket connection is closed, the following line might throw an error
499                self._socket.send(json.dumps({"type": "CloseStream"}))
500            except websockets.exceptions.ConnectionClosedOK as e:
501                self.logger.notice(f"_signal_exit  - connection closed: {e.code}")
502            except websockets.exceptions.WebSocketException as e:
503                self.logger.error(f"_signal_exit - WebSocketException: {str(e)}")
504            except Exception as e:
505                self.logger.error(f"_signal_exit - Exception: {str(e)}")
506
507            # push close event
508            try:
509                self._emit(
510                    LiveTranscriptionEvents.Close,
511                    CloseResponse(type=LiveTranscriptionEvents.Close.value),
512                )
513            except Exception as e:
514                self.logger.error(f"_signal_exit - Exception: {str(e)}")
515
516            # wait for task to send
517            time.sleep(0.5)
518
519        # signal exit
520        self._exit_event.set()
521
522        # closes the WebSocket connection gracefully
523        self.logger.verbose("clean up socket...")
524        if self._socket is not None:
525            self.logger.verbose("socket.wait_closed...")
526            try:
527                self._socket.close()
528                self._socket = None
529            except websockets.exceptions.WebSocketException as e:
530                self.logger.error("socket.wait_closed failed: %s", e)
