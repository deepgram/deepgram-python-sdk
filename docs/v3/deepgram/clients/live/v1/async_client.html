 93    async def start(
 94        self,
 95        options: Optional[Union[LiveOptions, Dict]] = None,
 96        addons: Optional[Dict] = None,
 97        headers: Optional[Dict] = None,
 98        members: Optional[Dict] = None,
 99        **kwargs,
100    ) -> bool:
101        """
102        Starts the WebSocket connection for live transcription.
103        """
104        self._logger.debug("AsyncLiveClient.start ENTER")
105        self._logger.info("options: %s", options)
106        self._logger.info("addons: %s", addons)
107        self._logger.info("headers: %s", headers)
108        self._logger.info("members: %s", members)
109        self._logger.info("kwargs: %s", kwargs)
110
111        if isinstance(options, LiveOptions) and not options.check():
112            self._logger.error("options.check failed")
113            self._logger.debug("AsyncLiveClient.start LEAVE")
114            raise DeepgramError("Fatal transcription options error")
115
116        self._addons = addons
117        self._headers = headers
118
119        # add "members" as members of the class
120        if members is not None:
121            self.__dict__.update(members)
122
123        # set kwargs as members of the class
124        if kwargs is not None:
125            self._kwargs = kwargs
126        else:
127            self._kwargs = {}
128
129        if isinstance(options, LiveOptions):
130            self._logger.info("LiveOptions switching class -> dict")
131            self._options = options.to_dict()
132        elif options is not None:
133            self._options = options
134        else:
135            self._options = {}
136
137        combined_options = self._options
138        if self._addons is not None:
139            self._logger.info("merging addons to options")
140            combined_options.update(self._addons)
141            self._logger.info("new options: %s", combined_options)
142        self._logger.debug("combined_options: %s", combined_options)
143
144        combined_headers = self._config.headers
145        if self._headers is not None:
146            self._logger.info("merging headers to options")
147            combined_headers.update(self._headers)
148            self._logger.info("new headers: %s", combined_headers)
149        self._logger.debug("combined_headers: %s", combined_headers)
150
151        url_with_params = append_query_params(self._websocket_url, combined_options)
152
153        try:
154            self._socket = await websockets.connect(
155                url_with_params,
156                extra_headers=combined_headers,
157                ping_interval=PING_INTERVAL,
158            )
159            self._exit_event.clear()
160
161            # debug the threads
162            for thread in threading.enumerate():
163                self._logger.debug("after running thread: %s", thread.name)
164            self._logger.debug("number of active threads: %s", threading.active_count())
165
166            # listen thread
167            self._listen_thread = asyncio.create_task(self._listening())
168
169            # keepalive thread
170            if self._config.is_keep_alive_enabled():
171                self._logger.notice("keepalive is enabled")
172                self._keep_alive_thread = asyncio.create_task(self._keep_alive())
173            else:
174                self._logger.notice("keepalive is disabled")
175
176            # flush thread
177            if self._config.is_auto_flush_enabled():
178                self._logger.notice("autoflush is enabled")
179                self._flush_thread = asyncio.create_task(self._flush())
180            else:
181                self._logger.notice("autoflush is disabled")
182
183            # debug the threads
184            for thread in threading.enumerate():
185                self._logger.debug("after running thread: %s", thread.name)
186            self._logger.debug("number of active threads: %s", threading.active_count())
187
188            # push open event
189            await self._emit(
190                LiveTranscriptionEvents(LiveTranscriptionEvents.Open),
191                OpenResponse(type=LiveTranscriptionEvents.Open),
192            )
193
194            self._logger.notice("start succeeded")
195            self._logger.debug("AsyncLiveClient.start LEAVE")
196            return True
197        except websockets.ConnectionClosed as e:
198            self._logger.error("ConnectionClosed in AsyncLiveClient.start: %s", e)
199            self._logger.debug("AsyncLiveClient.start LEAVE")
200            if self._config.options.get("termination_exception_connect") == "true":
201                raise
202            return False
203        except websockets.exceptions.WebSocketException as e:
204            self._logger.error("WebSocketException in AsyncLiveClient.start: %s", e)
205            self._logger.debug("AsyncLiveClient.start LEAVE")
206            if self._config.options.get("termination_exception_connect") == "true":
207                raise
208            return False
209        except Exception as e:  # pylint: disable=broad-except
210            self._logger.error("WebSocketException in AsyncLiveClient.start: %s", e)
211            self._logger.debug("AsyncLiveClient.start LEAVE")
212            if self._config.options.get("termination_exception_connect") == "true":
213                raise
214            return False

218    def on(self, event: LiveTranscriptionEvents, handler) -> None:
219        """
220        Registers event handlers for specific events.
221        """
222        self._logger.info("event subscribed: %s", event)
223        if event in LiveTranscriptionEvents.__members__.values() and callable(handler):
224            self._event_handlers[event].append(handler)

226    # triggers the registered event handlers for a specific event
227    async def _emit(self, event: LiveTranscriptionEvents, *args, **kwargs) -> None:
228        """
229        Emits events to the registered event handlers.
230        """
231        self._logger.debug("AsyncLiveClient._emit ENTER")
232        self._logger.debug("callback handlers for: %s", event)
233
234        # debug the threads
235        for thread in threading.enumerate():
236            self._logger.debug("after running thread: %s", thread.name)
237        self._logger.debug("number of active threads: %s", threading.active_count())
238
239        tasks = []
240        for handler in self._event_handlers[event]:
241            tasks.append(asyncio.create_task(handler(self, *args, **kwargs)))
242
243        if len(tasks) > 0:
244            self._logger.debug("waiting for tasks to finish...")
245            await asyncio.gather(*filter(None, tasks), return_exceptions=True)
246
247        # debug the threads
248        for thread in threading.enumerate():
249            self._logger.debug("after running thread: %s", thread.name)
250        self._logger.debug("number of active threads: %s", threading.active_count())
251
252        self._logger.debug("AsyncLiveClient._emit LEAVE")

255    async def _listening(self) -> None:
256        """
257        Listens for messages from the WebSocket connection.
258        """
259        self._logger.debug("AsyncLiveClient._listening ENTER")
260
261        while True:
262            try:
263                if self._exit_event.is_set():
264                    self._logger.notice("_listening exiting gracefully")
265                    self._logger.debug("AsyncLiveClient._listening LEAVE")
266                    return
267
268                if self._socket is None:
269                    self._logger.warning("socket is empty")
270                    self._logger.debug("AsyncLiveClient._listening LEAVE")
271                    return
272
273                message = str(await self._socket.recv())
274
275                if message is None:
276                    self._logger.spam("message is None")
277                    continue
278
279                data = json.loads(message)
280                response_type = data.get("type")
281                self._logger.debug("response_type: %s, data: %s", response_type, data)
282
283                match response_type:
284                    case LiveTranscriptionEvents.Open:
285                        open_result: OpenResponse = OpenResponse.from_json(message)
286                        self._logger.verbose("OpenResponse: %s", open_result)
287                        await self._emit(
288                            LiveTranscriptionEvents(LiveTranscriptionEvents.Open),
289                            open=open_result,
290                            **dict(cast(Dict[Any, Any], self._kwargs)),
291                        )
292                    case LiveTranscriptionEvents.Transcript:
293                        msg_result: LiveResultResponse = LiveResultResponse.from_json(
294                            message
295                        )
296                        self._logger.verbose("LiveResultResponse: %s", msg_result)
297
298                        # auto flush
299                        if self._config.is_inspecting_messages():
300                            inspect_res = await self._inspect(msg_result)
301                            if not inspect_res:
302                                self._logger.error("inspect_res failed")
303
304                        await self._emit(
305                            LiveTranscriptionEvents(LiveTranscriptionEvents.Transcript),
306                            result=msg_result,
307                            **dict(cast(Dict[Any, Any], self._kwargs)),
308                        )
309                    case LiveTranscriptionEvents.Metadata:
310                        meta_result: MetadataResponse = MetadataResponse.from_json(
311                            message
312                        )
313                        self._logger.verbose("MetadataResponse: %s", meta_result)
314                        await self._emit(
315                            LiveTranscriptionEvents(LiveTranscriptionEvents.Metadata),
316                            metadata=meta_result,
317                            **dict(cast(Dict[Any, Any], self._kwargs)),
318                        )
319                    case LiveTranscriptionEvents.SpeechStarted:
320                        ss_result: SpeechStartedResponse = (
321                            SpeechStartedResponse.from_json(message)
322                        )
323                        self._logger.verbose("SpeechStartedResponse: %s", ss_result)
324                        await self._emit(
325                            LiveTranscriptionEvents(
326                                LiveTranscriptionEvents.SpeechStarted
327                            ),
328                            speech_started=ss_result,
329                            **dict(cast(Dict[Any, Any], self._kwargs)),
330                        )
331                    case LiveTranscriptionEvents.UtteranceEnd:
332                        ue_result: UtteranceEndResponse = (
333                            UtteranceEndResponse.from_json(message)
334                        )
335                        self._logger.verbose("UtteranceEndResponse: %s", ue_result)
336                        await self._emit(
337                            LiveTranscriptionEvents(
338                                LiveTranscriptionEvents.UtteranceEnd
339                            ),
340                            utterance_end=ue_result,
341                            **dict(cast(Dict[Any, Any], self._kwargs)),
342                        )
343                    case LiveTranscriptionEvents.Close:
344                        close_result: CloseResponse = CloseResponse.from_json(message)
345                        self._logger.verbose("CloseResponse: %s", close_result)
346                        await self._emit(
347                            LiveTranscriptionEvents(LiveTranscriptionEvents.Close),
348                            close=close_result,
349                            **dict(cast(Dict[Any, Any], self._kwargs)),
350                        )
351                    case LiveTranscriptionEvents.Error:
352                        err_error: ErrorResponse = ErrorResponse.from_json(message)
353                        self._logger.verbose("ErrorResponse: %s", err_error)
354                        await self._emit(
355                            LiveTranscriptionEvents(LiveTranscriptionEvents.Error),
356                            error=err_error,
357                            **dict(cast(Dict[Any, Any], self._kwargs)),
358                        )
359                    case _:
360                        self._logger.warning(
361                            "Unknown Message: response_type: %s, data: %s",
362                            response_type,
363                            data,
364                        )
365                        unhandled_error: UnhandledResponse = UnhandledResponse(
366                            type=LiveTranscriptionEvents(
367                                LiveTranscriptionEvents.Unhandled
368                            ),
369                            raw=message,
370                        )
371                        await self._emit(
372                            LiveTranscriptionEvents(LiveTranscriptionEvents.Unhandled),
373                            unhandled=unhandled_error,
374                            **dict(cast(Dict[Any, Any], self._kwargs)),
375                        )
376
377            except websockets.exceptions.ConnectionClosedOK as e:
378                self._logger.notice(f"_listening({e.code}) exiting gracefully")
379                self._logger.debug("AsyncLiveClient._listening LEAVE")
380                return
381
382            except websockets.exceptions.ConnectionClosed as e:
383                if e.code == 1000:
384                    self._logger.notice(f"_listening({e.code}) exiting gracefully")
385                    self._logger.debug("AsyncLiveClient._listening LEAVE")
386                    return
387
388                self._logger.error(
389                    "ConnectionClosed in AsyncLiveClient._listening with code %s: %s",
390                    e.code,
391                    e.reason,
392                )
393                cc_error: ErrorResponse = ErrorResponse(
394                    "ConnectionClosed in AsyncLiveClient._listening",
395                    f"{e}",
396                    "ConnectionClosed",
397                )
398                await self._emit(
399                    LiveTranscriptionEvents(LiveTranscriptionEvents.Error),
400                    error=cc_error,
401                    **dict(cast(Dict[Any, Any], self._kwargs)),
402                )
403
404                # signal exit and close
405                await self._signal_exit()
406
407                self._logger.debug("AsyncLiveClient._listening LEAVE")
408
409                if self._config.options.get("termination_exception") == "true":
410                    raise
411                return
412
413            except websockets.exceptions.WebSocketException as e:
414                self._logger.error(
415                    "WebSocketException in AsyncLiveClient._listening: %s", e
416                )
417                ws_error: ErrorResponse = ErrorResponse(
418                    "WebSocketException in AsyncLiveClient._listening",
419                    f"{e}",
420                    "WebSocketException",
421                )
422                await self._emit(
423                    LiveTranscriptionEvents(LiveTranscriptionEvents.Error),
424                    error=ws_error,
425                    **dict(cast(Dict[Any, Any], self._kwargs)),
426                )
427
428                # signal exit and close
429                await self._signal_exit()
430
431                self._logger.debug("AsyncLiveClient._listening LEAVE")
432
433                if self._config.options.get("termination_exception") == "true":
434                    raise
435                return
436
437            except Exception as e:  # pylint: disable=broad-except
438                self._logger.error("Exception in AsyncLiveClient._listening: %s", e)
439                e_error: ErrorResponse = ErrorResponse(
440                    "Exception in AsyncLiveClient._listening",
441                    f"{e}",
442                    "Exception",
443                )
444                await self._emit(
445                    LiveTranscriptionEvents(LiveTranscriptionEvents.Error),
446                    error=e_error,
447                    **dict(cast(Dict[Any, Any], self._kwargs)),
448                )
449
450                # signal exit and close
451                await self._signal_exit()
452
453                self._logger.debug("AsyncLiveClient._listening LEAVE")
454
455                if self._config.options.get("termination_exception") == "true":
456                    raise
457                return

574    async def _flush(self) -> None:
575        self._logger.debug("AsyncLiveClient._flush ENTER")
576
577        delta_in_ms_str = self._config.options.get("auto_flush_reply_delta")
578        if delta_in_ms_str is None:
579            self._logger.error("auto_flush_reply_delta is None")
580            self._logger.debug("AsyncLiveClient._flush LEAVE")
581            return
582        delta_in_ms = float(delta_in_ms_str)
583
584        while True:
585            try:
586                await asyncio.sleep(HALF_SECOND)
587
588                if self._exit_event.is_set():
589                    self._logger.notice("_flush exiting gracefully")
590                    self._logger.debug("AsyncLiveClient._flush LEAVE")
591                    return
592
593                if self._socket is None:
594                    self._logger.notice("socket is None, exiting flush")
595                    self._logger.debug("AsyncLiveClient._flush LEAVE")
596                    return
597
598                if self._last_datagram is None:
599                    self._logger.debug("AutoFlush last_datagram is None")
600                    continue
601
602                delta = datetime.now() - self._last_datagram
603                diff_in_ms = delta.total_seconds() * 1000
604                self._logger.debug("AutoFlush delta: %f", diff_in_ms)
605                if diff_in_ms < delta_in_ms:
606                    self._logger.debug("AutoFlush delta is less than threshold")
607                    continue
608
609                self._last_datagram = None
610                await self.finalize()
611
612            except websockets.exceptions.ConnectionClosedOK as e:
613                self._logger.notice(f"_flush({e.code}) exiting gracefully")
614                self._logger.debug("AsyncLiveClient._flush LEAVE")
615                return
616
617            except websockets.exceptions.ConnectionClosed as e:
618                if e.code == 1000:
619                    self._logger.notice(f"_flush({e.code}) exiting gracefully")
620                    self._logger.debug("AsyncLiveClient._flush LEAVE")
621                    return
622
623                self._logger.error(
624                    "ConnectionClosed in AsyncLiveClient._flush with code %s: %s",
625                    e.code,

627                )

629                    "ConnectionClosed in AsyncLiveClient._flush",
630                    f"{e}",

632                )
633                await self._emit(
634                    LiveTranscriptionEvents(LiveTranscriptionEvents.Error),
635                    error=cc_error,
636                    **dict(cast(Dict[Any, Any], self._kwargs)),
637                )
638
639                # signal exit and close

641

643

645                    raise
