231
232        if api_key == "" and config is not None:
233            self._logger.info("Attempting to set API key from config object")
234            api_key = config.api_key
235        if api_key == "":
236            self._logger.info("Attempting to set API key from environment variable")
237            api_key = os.getenv("DEEPGRAM_API_KEY", "")
238        if api_key == "":
239            self._logger.warning("WARNING: API key is missing")
240
241        self.api_key = api_key
242        if config is None:  # Use default configuration
243            self._config = DeepgramClientOptions(self.api_key)
244        else:
245            config.set_apikey(self.api_key)
246            self._config = config

366        def v(self, version: str = ""):
367            # pylint: disable-msg=too-many-statements
368            """
369            Returns a client for the specified version of the API.
370            """
371            self._logger.debug("Version.v ENTER")
372            self._logger.info("version: %s", version)
373            if len(version) == 0:
374                self._logger.error("version is empty")
375                self._logger.debug("Version.v LEAVE")
376                raise DeepgramModuleError("Invalid module version")
377
378            parent = ""
379            filename = ""
380            classname = ""
381            match self._parent:
382                case "manage":
383                    parent = "manage"
384                    filename = "client"
385                    classname = "ManageClient"
386                case "asyncmanage":
387                    parent = "manage"
388                    filename = "async_client"
389                    classname = "AsyncManageClient"
390                case "asyncspeak":
391                    return AsyncSpeakRESTClient(self._config)
392                case "selfhosted":
393                    parent = "selfhosted"
394                    filename = "client"
395                    classname = "SelfHostedClient"
396                case "asyncselfhosted":
397                    parent = "selfhosted"
398                    filename = "async_client"
399                    classname = "AsyncSelfHostedClient"
400                case _:
401                    self._logger.error("parent unknown: %s", self._parent)
402                    self._logger.debug("Version.v LEAVE")
403                    raise DeepgramModuleError("Invalid parent type")
404
405            # create class path
406            path = f"deepgram.clients.{parent}.v{version}.{filename}"
407            self._logger.info("path: %s", path)
408            self._logger.info("classname: %s", classname)
409
410            # import class
411            mod = import_module(path)
412            if mod is None:
413                self._logger.error("module path is None")
414                self._logger.debug("Version.v LEAVE")
415                raise DeepgramModuleError("Unable to find package")
416
417            my_class = getattr(mod, classname)
418            if my_class is None:
419                self._logger.error("my_class is None")
420                self._logger.debug("Version.v LEAVE")
421                raise DeepgramModuleError("Unable to find class")
422
423            # instantiate class
424            my_class_instance = my_class(self._config)
425            self._logger.notice("Version.v succeeded")
426            self._logger.debug("Version.v LEAVE")
427            return my_class_instance
