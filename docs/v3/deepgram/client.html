346    """
347    Represents a client for interacting with the Deepgram API.
348
349    This class provides a client for making requests to the Deepgram API with various configuration options.
350
351    Attributes:
352        api_key (str): The Deepgram API key used for authentication.
353        config_options (DeepgramClientOptions): An optional configuration object specifying client options.
354
355    Raises:
356        DeepgramApiKeyError: If the API key is missing or invalid.
357
358    Methods:
359        listen: Returns a ListenClient instance for interacting with Deepgram's transcription services.
360
361        manage: (Preferred) Returns a Threaded ManageClient instance for managing Deepgram resources.
362        selfhosted: (Preferred) Returns an Threaded SelfHostedClient instance for interacting with Deepgram's on-premises API.
363
364        asyncmanage: Returns an (Async) ManageClient instance for managing Deepgram resources.
365        asyncselfhosted: Returns an (Async) SelfHostedClient instance for interacting with Deepgram's on-premises API.
366    """
367
368    _config: DeepgramClientOptions
369    _logger: verboselogs.VerboseLogger
370
371    def __init__(
372        self,
373        api_key: str = "",
374        config: Optional[DeepgramClientOptions] = None,
375    ):
376        self._logger = verboselogs.VerboseLogger(__name__)
377        self._logger.addHandler(logging.StreamHandler())
378
379        if api_key == "" and config is not None:
380            self._logger.info("Attempting to set API key from config object")
381            api_key = config.api_key
382        if api_key == "":
383            self._logger.info("Attempting to set API key from environment variable")
384            api_key = os.getenv("DEEPGRAM_API_KEY", "")
385        if api_key == "":
386            self._logger.warning("WARNING: API key is missing")
387
388        self.api_key = api_key
389        if config is None:  # Use default configuration
390            self._config = DeepgramClientOptions(self.api_key)
391        else:
392            config.set_apikey(self.api_key)
393            self._config = config
394

397        """

485        """
486        Represents a version of the Deepgram API.
487        """
488
489        _logger: verboselogs.VerboseLogger
490        _config: DeepgramClientOptions
491        _parent: str
492
493        def __init__(self, config, parent: str):
494            self._logger = verboselogs.VerboseLogger(__name__)
495            self._logger.addHandler(logging.StreamHandler())
496            self._logger.setLevel(config.verbose)
497
498            self._config = config
499            self._parent = parent
500
501        # FUTURE VERSIONING:
502        # When v2 or v1.1beta1 or etc. This allows easy access to the latest version of the API.
503        # @property
504        # def latest(self):
505        #     match self._parent:
506        #         case "manage":
507        #             return ManageClient(self._config)
508        #         case "selfhosted":
509        #             return SelfHostedClient(self._config)
510        #         case _:
511        #             raise DeepgramModuleError("Invalid parent")
512
513        def v(self, version: str = ""):
514            # pylint: disable-msg=too-many-statements
515            """
516            Returns a client for the specified version of the API.
517            """
518            self._logger.debug("Version.v ENTER")
519            self._logger.info("version: %s", version)
520            if len(version) == 0:
521                self._logger.error("version is empty")
522                self._logger.debug("Version.v LEAVE")
523                raise DeepgramModuleError("Invalid module version")
524
525            parent = ""
526            filename = ""
527            classname = ""
528            match self._parent:
529                case "manage":
530                    parent = "manage"
531                    filename = "client"
532                    classname = "ManageClient"
533                case "asyncmanage":
534                    parent = "manage"
535                    filename = "async_client"
536                    classname = "AsyncManageClient"
537                case "asyncspeak":
538                    return AsyncSpeakRESTClient(self._config)
539                case "selfhosted":
540                    parent = "selfhosted"
541                    filename = "client"
542                    classname = "SelfHostedClient"
543                case "asyncselfhosted":
544                    parent = "selfhosted"
545                    filename = "async_client"
546                    classname = "AsyncSelfHostedClient"
547                case _:
548                    self._logger.error("parent unknown: %s", self._parent)
549                    self._logger.debug("Version.v LEAVE")
550                    raise DeepgramModuleError("Invalid parent type")
551
552            # create class path
553            path = f"deepgram.clients.{parent}.v{version}.{filename}"
554            self._logger.info("path: %s", path)
555            self._logger.info("classname: %s", classname)
556
557            # import class
558            mod = import_module(path)
559            if mod is None:
560                self._logger.error("module path is None")
561                self._logger.debug("Version.v LEAVE")
562                raise DeepgramModuleError("Unable to find package")
563
564            my_class = getattr(mod, classname)
565            if my_class is None:
566                self._logger.error("my_class is None")
567                self._logger.debug("Version.v LEAVE")
568                raise DeepgramModuleError("Unable to find class")
569
570            # instantiate class
571            my_class_instance = my_class(self._config)
572            self._logger.notice("Version.v succeeded")
573            self._logger.debug("Version.v LEAVE")
574            return my_class_instance
